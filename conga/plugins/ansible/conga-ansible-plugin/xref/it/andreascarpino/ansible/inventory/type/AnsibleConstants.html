19  package it.andreascarpino.ansible.inventory.type;
20  
21  /**
22   * @author Andrea Scarpino
23   */
24  public class AnsibleConstants {
25  
26    /**
27     * Connection type to the host. Candidates are <code>local</code>,
28     * <code>smart</code>, <code>ssh</code>, <code>paramiko</code> or
29     * <code>winrm</code>. The default is <code>smart</code>.
30     *
31     * @see AnsibleConnection
32     */
33    public static final String ANSIBLE_CONNECTION = "ansible_connection";
34  
35    /**
36     * The name of the host to connect to, if different from the alias you wish
37     * to give to it.
38     */
39    public static final String ANSIBLE_HOST = "ansible_host";
40  
41    /**
42     * The ssh port number, if not 22.
43     */
44    public static final String ANSIBLE_PORT = "ansible_port";
45  
46    /**
47     * The default ssh user name to use.
48     */
49    public static final String ANSIBLE_USER = "ansible_user";
50  
51    /**
52     * The user password to use (this is insecure, we strongly recommend using
53     * <code>--ask-pass</code> or SSH keys).
54     */
55    public static final String ANSIBLE_PASSWORD = "ansible_password";
56  
57    /**
58     * Private key file used by ssh. Useful if using multiple keys and you don't
59     * want to use SSH agent.
60     */
61    public static final String ANSIBLE_SSH_PRIVATE_KEY_FILE = "ansible_ssh_private_key_file";
62  
63    /**
64     * This setting is always appended to the default command line for sftp,
65     * scp, and ssh. Useful to configure a <code>ProxyCommand</code> for a
66     * certain host (or group).
67     */
68    public static final String ANSIBLE_SSH_COMMON_ARGS = "ansible_ssh_common_args";
69  
70    /**
71     * This setting is always appended to the default sftp command line.
72     */
73    public static final String ANSIBLE_SFTP_EXTRA_ARGS = "ansible_sftp_extra_args";
74  
75    /**
76     * This setting is always appended to the default scp command line.
77     */
78    public static final String ANSIBLE_SCP_EXTRA_ARGS = "ansible_scp_extra_args";
79  
80    /**
81     * This setting is always appended to the default ssh command line.
82     */
83    public static final String ANSIBLE_SSH_EXTRA_ARGS = "ansible_ssh_extra_args";
84  
85    /**
86     * Determines whether or not to use SSH pipelining. This can override the
87     * <code>pipelining</code> setting in <code>ansible.cfg</code>.
88     */
89    public static final String ANSIBLE_SSH_PIPELINING = "ansible_ssh_pipelining";
90  
91    /**
92     * Equivalent to <code>ansible_sudo</code> or <code>ansible_su</code>,
93     * allows to force privilege escalation.
94     */
95    public static final String ANSIBLE_BECOME = "ansible_become";
96  
97    /**
98     * Allows to set privilege escalation method.
99     */
100   public static final String ANSIBLE_BECOME_METHOD = "ansible_become_method";
101 
102   /**
103    * Equivalent to <code>ansible_sudo_user</code> or
104    * <code>ansible_su_user</code>, allows to set the user you become through
105    * privilege escalation.
106    */
107   public static final String ANSIBLE_BECOME_USER = "ansible_become_user";
108 
109   /**
110    * Equivalent to <code>ansible_sudo_pass</code> or
111    * <code>ansible_su_pass</code>, allows you to set the privilege escalation
112    * password.
113    */
114   public static final String ANSIBLE_BECOME_PASS = "ansible_become_pass";
115 
116   /**
117    * The shell type of the target system. Commands are formatted using
118    * 'sh'-style syntax by default. Setting this to <code>csh</code> or
119    * <code>fish</code> will cause commands executed on target systems to
120    * follow those shell's syntax instead.
121    */
122   public static final String ANSIBLE_SHELL_TYPE = "ansible_shell_type";
123 
124   /**
125    * The target host python path. This is useful for systems with more than
126    * one Python or not located at <code>/usr/bin/python</code> such as BSD, or
127    * where <code>/usr/bin/python</code> is not a 2.X series Python. We do not
128    * use the <code>/usr/bin/env</code> mechanism as that requires the remote
129    * user's path to be set right and also assumes the <code>python</code>
130    * executable is named python, where the executable might be named something
131    * like <code>python26</code>.
132    */
133   public static final String ANSIBLE_PYTHON_INTERPRETER = "ansible_python_interpreter";
134 
135   /**
136    * Specify the connection scheme (<code>http</code> or <code>https</code>)
137    * to use for the WinRM connection. Ansible uses <code>https</code> by
138    * default unless the port is 5985.
139    *
140    * @see AnsibleWinRMScheme
141    */
142   public static final String ANSIBLE_WINRM_SCHEME = "ansible_winrm_scheme";
143 
144   /**
145    * Specify an alternate path to the WinRM endpoint. Ansible uses
146    * <code>/wsman</code> by default.
147    */
148   public static final String ANSIBLE_WINRM_PATH = "ansible_winrm_path";
149 
150   /**
151    * Specify the realm to use for Kerberos authentication. If the username
152    * contains "@", Ansible will use the part of the username after "@" by
153    * default.
154    */
155   public static final String ANSIBLE_WINRM_REALM = "ansible_winrm_realm";
156 
157   /**
158    * Specify one or more transports as a comma-separated list. By default,
159    * Ansible will use <code>kerberos,plaintext</code> if the "kerberos" module
160    * is installed and a realm is defined, otherwise <code>plaintext</code>.
161    */
162   public static final String ANSIBLE_WINRM_TRANSPORT = "ansible_winrm_transport";
163 
164   /**
165    * Specify the server certificate validation mode (<code>ignore</code> or
166    * <code>validate</code>). Ansible defaults to <code>validate</code> on
167    * Python 2.7.9 and higher, which will result in certificate validation
168    * errors against the Windows self-signed certificates. Unless verifiable
169    * certificates have been configured on the WinRM listeners, this should be
170    * set to <code>ignore</code>.
171    */
172   public static final String ANSIBLE_WINRM_SERVER_CERT_VALIDATION = "ansible_winrm_server_cert_validation";
173 
174   /**
175    * Set to true to enable delegation of commands on the remote host when
176    * using kerberos.
177    */
178   public static final String ANSIBLE_WINRM_KERBEROS_DELEGATION = "ansible_winrm_kerberos_delegation";
179 
180   /**
181    * Maximum seconds to wait before an HTTP connect/read times out (default
182    * 30). This value should be slightly higher than
183    * <code>operation_timeout_sec</code>, as the server can block *at least*
184    * that long.
185    */
186   public static final String ANSIBLE_WINRM_READ_TIMEOUT_SEC = "ansible_winrm_read_timeout_sec";
187 
188   /**
189    * Maximum allowed time in seconds for any single wsman HTTP operation
190    * (default 20). Note that operation timeouts while receiving output (the
191    * only wsman operation that should take any significant time, and where
192    * these timeouts are expected) will be silently retried indefinitely.
193    */
194   public static final String ANSIBLE_WINRM_OPERATION_TIMEOUT_SEC = "ansible_winrm_operation_timeout_sec";
195 
196   /**
197    * The hostname to use for the kerberos exchange (defaults to the hostname
198    * in the endpoint URL).
199    */
200   public static final String ANSIBLE_WINRM_KERBEROS_HOSTNAME_OVERRIDE = "ansible_winrm_kerberos_hostname_override";
201 
202   /**
203    * Default group that contains all hosts in the inventory.
204    */
205   public static final String ALL_GROUP = "all";
206

208   }
