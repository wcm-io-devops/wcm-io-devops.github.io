36    public static AnsibleInventory read(String text) {
37      final AnsibleInventorype/AnsibleInventory.html#AnsibleInventory">AnsibleInventory inventory = new AnsibleInventory();
38  
39      final StringTokenizer tokenizer = new StringTokenizer(text, " \t\n\r\f", true);
40  
41      AnsibleGroup group = null;
42      AnsibleHost host = null;
43      boolean skipComment = false;
44      boolean isVarsBlock = false;
45      boolean isChildrenBlock = false;
46      while (tokenizer.hasMoreTokens()) {
47        final String token = tokenizer.nextToken();
48  
49        // New line, reset the comment flag
50        if ("\n".equals(token)) {
51          skipComment = false;
52          continue;
53        }
54  
55        // We are still reading a comment line
56        if (skipComment) {
57          continue;
58        }
59  
60        // Ignore separators
61        if (" ".equals(token) || "\t".equals(token) || "\r".equals(token) || "\f".equals(token)) {
62          continue;
63        }
64  
65        // We are reading a comment
66        if (token.startsWith(";") || token.startsWith("#")) {
67          skipComment = true;
68          continue;
69        }
70  
71        if (token.startsWith("[")) {
72          host = null;
73          isChildrenBlock = false;
74          isVarsBlock = false;
75  
76          String groupName = token.replaceAll("^\\[", "").replaceAll("]$", "");
77  
78          if (groupName.contains(":")) {
79            final String[] g = groupName.split(":");
80  
81            groupName = g[0];
82  
83            if ("vars".equals(g[1])) {
84              isVarsBlock = true;
85              group = inventory.getGroup(groupName);
86            } else if ("children".equals(g[1])) {
87              isChildrenBlock = true;
88              group = new AnsibleGroup(groupName);
89              inventory.addGroup(group);
90            }
91          } else {
92            group = new AnsibleGroup(groupName);
93            inventory.addGroup(group);
94          }
95        } else if (token.contains("=")) {
96          final String[] v = token.split("=");
97          // Replace YAML backslashes escapes
98          final AnsibleVariabletype/AnsibleVariable.html#AnsibleVariable">AnsibleVariable variable = new AnsibleVariable(v[0], v[1].replace("\\\\", "\\"));
99  
100         if (host != null) {
101           host.addVariable(variable);
102         } else if (isVarsBlock && group != null) {
103           for (AnsibleGroup s : group.getSubgroups()) {
104             for (AnsibleHost h : s.getHosts()) {
105               h.addVariable(variable);
106             }
107           }
108           for (AnsibleHost h : group.getHosts()) {
109             h.addVariable(variable);
110           }
111         }
112       } else {
113         if (group == null) {
114           host = new AnsibleHost(token);
115           inventory.addHost(host);
116         } else if (isChildrenBlock) {
117           final AnsibleGroup g = inventory.getGroup(token);
118           if (g != null) {
119             group.addSubgroup(g);
120           } else {
121             group.addSubgroup(new AnsibleGroup(token));
122           }
123         } else {
124           host = new AnsibleHost(token);
125           group.addHost(host);
126         }
127       }
128     }
129 
130     return inventory;
131   }
