52  /**
53   * The <code>ConfigurationHandler</code> class implements configuration reading
54   * form a <code>java.io.InputStream</code> and writing to a
55   * <code>java.io.OutputStream</code> on behalf of the
56   * {@link FilePersistenceManager} class.
57   *
58   * <pre>
59   * cfg = prop &quot;=&quot; value .
60   *  prop = symbolic-name . // 1.4.2 of OSGi Core Specification
61   *  symbolic-name = token { &quot;.&quot; token } .
62   *  token = { [ 0..9 ] | [ a..z ] | [ A..Z ] | '_' | '-' } .
63   *  value = [ type ] ( &quot;[&quot; values &quot;]&quot; | &quot;(&quot; values &quot;)&quot; | simple ) .
64   *  values = simple { &quot;,&quot; simple } .
65   *  simple = &quot;&quot;&quot; stringsimple &quot;&quot;&quot; .
66   *  type = // 1-char type code .
67   *  stringsimple = // quoted string representation of the value .
68   * </pre>
69   */

196   /**
197    * Writes the configuration data from the <code>Dictionary</code> to the
198    * given <code>OutputStream</code>.
199    * <p>
200    * This method writes at the current location in the stream and does not
201    * close the outputstream.
202    * @param out
203    *          The <code>OutputStream</code> to write the configurtion data
204    *          to.
205    * @param properties
206    *          The <code>Dictionary</code> to write.
207    * @throws IOException
208    *           If an error occurrs writing to the output stream.
209    */
210   public static void write(OutputStream out, Dictionary properties) throws IOException {
211     BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(out, ENCODING));
212 
213     for (Enumeration ce = properties.keys(); ce.hasMoreElements();) {
214       String key = (String)ce.nextElement();
215 
216       // cfg = prop "=" value "." .
217       writeQuoted(bw, key);
218       bw.write(TOKEN_EQ);
219       writeValue(bw, properties.get(key));
220       bw.write(CRLF);
221     }
222 
223     bw.flush();
224   }

228    * Reads configuration data from the given <code>InputStream</code> and
229    * returns a new <code>Dictionary</code> object containing the data.
230    * <p>
231    * This method reads from the current location in the stream upto the end of
232    * the stream but does not close the stream at the end.
233    * @param ins
234    *          The <code>InputStream</code> from which to read the
235    *          configuration data.
236    * @return A <code>Dictionary</code> object containing the configuration
237    *         data. This object may be empty if the stream contains no
238    *         configuration data.
239    * @throws IOException
240    *           If an error occurrs reading from the stream. This exception
241    *           is also thrown if a syntax error is encountered.
242    */
243   public static Dictionary read(InputStream ins) throws IOException {
244     return new ConfigurationHandler_ConfigAdmin184().readInternal(ins);
245   }

260   private Dictionary readInternal(InputStream ins) throws IOException {
261     BufferedReader br = new BufferedReader(new InputStreamReader(ins, ENCODING));
262     PushbackReader pr = new PushbackReader(br, 1);
263 
264     token = 0;
265     tokenValue = null;
266     line = 0;
267     pos = 0;
268 
269     Hashtable configuration = new Hashtable();
270     token = 0;
271     while (nextToken(pr) == TOKEN_NAME) {
272       String key = tokenValue;
273 
274       // expect equal sign
275       if (nextToken(pr) != TOKEN_EQ) {
276         throw readFailure(token, TOKEN_EQ);
277       }
278 
279       // expect the token value
280       Object value = readValue(pr);
281       if (value != null) {
282         configuration.put(key, value);
283       }
284     }
285 
286     return configuration;
287   }

290   /**
291    * value = type ( "[" values "]" | "(" values ")" | simple ) . values =
292    * value { "," value } . simple = "{" stringsimple "}" . type = // 1-char
293    * type code . stringsimple = // quoted string representation of the value .
294    * @param pr
295    * @return
296    * @throws IOException
297    */
298   private Object readValue(PushbackReader pr) throws IOException {
299     // read (optional) type code
300     int type = read(pr);
301 
302     // read value kind code if type code is not a value kinde code
303     int code;
304     if (code2Type.containsKey(new Integer(type))) {
305       code = read(pr);
306     }
307     else {
308       code = type;
309       type = TOKEN_SIMPLE_STRING;
310     }
311 
312     switch (code) {
313       case TOKEN_ARR_OPEN:
314         return readArray(type, pr);
315 
316       case TOKEN_VEC_OPEN:
317         return readCollection(type, pr);
318 
319       case TOKEN_VAL_OPEN:
320         Object value = readSimple(type, pr);
321         ensureNext(pr, TOKEN_VAL_CLOS);
322         return value;
323 
324       default:
325         return null;
326     }
327   }

330   private Object readArray(int typeCode, PushbackReader pr) throws IOException {
331     List list = new ArrayList();
332     for (;;) {
333       int c = read(pr);
334       if (c == TOKEN_VAL_OPEN) {
335         Object value = readSimple(typeCode, pr);
336         if (value == null) {
337           // abort due to error
338           return null;
339         }
340 
341         ensureNext(pr, TOKEN_VAL_CLOS);
342 
343         list.add(value);
344 
345         c = read(pr);
346       }
347 
348       if (c == TOKEN_ARR_CLOS) {
349         Class type = (Class)code2Type.get(new Integer(typeCode));
350         Object array = Array.newInstance(type, list.size());
351         for (int i = 0; i < list.size(); i++) {
352           Array.set(array, i, list.get(i));
353         }
354         return array;
355       }
356       else if (c < 0) {
357         return null;
358       }
359       else if (c != TOKEN_COMMA) {
360         return null;
361       }
362     }
363   }

366   private Collection readCollection(int typeCode, PushbackReader pr) throws IOException {
367     Collection collection = new ArrayList();
368     for (;;) {
369       int c = read(pr);
370       if (c == TOKEN_VAL_OPEN) {
371         Object value = readSimple(typeCode, pr);
372         if (value == null) {
373           // abort due to error
374           return null;
375         }
376 
377         ensureNext(pr, TOKEN_VAL_CLOS);
378 
379         collection.add(value);
380 
381         c = read(pr);
382       }
383 
384       if (c == TOKEN_VEC_CLOS) {
385         return collection;
386       }
387       else if (c < 0) {
388         return null;
389       }
390       else if (c != TOKEN_COMMA) {
391         return null;
392       }
393     }
394   }

397   private Object readSimple(int code, PushbackReader pr) throws IOException {
398     switch (code) {
399       case -1:
400         return null;
401 
402       case TOKEN_SIMPLE_STRING:
403         return readQuoted(pr);
404 
405       // Simple/Primitive, only use wrapper classes
406       case TOKEN_SIMPLE_INTEGER:
407       case TOKEN_PRIMITIVE_INT:
408         return Integer.valueOf(readQuoted(pr));
409 
410       case TOKEN_SIMPLE_LONG:
411       case TOKEN_PRIMITIVE_LONG:
412         return Long.valueOf(readQuoted(pr));
413 
414       case TOKEN_SIMPLE_FLOAT:
415       case TOKEN_PRIMITIVE_FLOAT:
416         int fBits = Integer.parseInt(readQuoted(pr));
417         return new Float(Float.intBitsToFloat(fBits));
418 
419       case TOKEN_SIMPLE_DOUBLE:
420       case TOKEN_PRIMITIVE_DOUBLE:
421         long dBits = Long.parseLong(readQuoted(pr));
422         return new Double(Double.longBitsToDouble(dBits));
423 
424       case TOKEN_SIMPLE_BYTE:
425       case TOKEN_PRIMITIVE_BYTE:
426         return Byte.valueOf(readQuoted(pr));
427 
428       case TOKEN_SIMPLE_SHORT:
429       case TOKEN_PRIMITIVE_SHORT:
430         return Short.valueOf(readQuoted(pr));
431 
432       case TOKEN_SIMPLE_CHARACTER:
433       case TOKEN_PRIMITIVE_CHAR:
434         String cString = readQuoted(pr);
435         if (cString != null && cString.length() > 0) {
436           return new Character(cString.charAt(0));
437         }
438         return null;
439 
440       case TOKEN_SIMPLE_BOOLEAN:
441       case TOKEN_PRIMITIVE_BOOLEAN:
442         return Boolean.valueOf(readQuoted(pr));
443 
444       // unknown type code
445       default:
446         return null;
447     }
448   }

451   private void ensureNext(PushbackReader pr, int expected) throws IOException {
452     int next = read(pr);
453     if (next != expected) {
454       readFailure(next, expected);
455     }
456   }

459   private boolean checkNext(PushbackReader pr, int expected) throws IOException {
460     int next = read(pr);
461     if (next < 0) {
462       return false;
463     }
464 
465     if (next == expected) {
466       return true;
467     }
468 
469     return false;
470   }

473   private String readQuoted(PushbackReader pr) throws IOException {
474     StringBuffer buf = new StringBuffer();
475     for (;;) {
476       int c = read(pr);
477       switch (c) {
478         // escaped character
479         case '\\':
480           c = read(pr);
481           switch (c) {
482             // well known escapes
483             case 'b':
484               buf.append('\b');
485               break;
486             case 't':
487               buf.append('\t');
488               break;
489             case 'n':
490               buf.append('\n');
491               break;
492             case 'f':
493               buf.append('\f');
494               break;
495             case 'r':
496               buf.append('\r');
497               break;
498             case 'u':// need 4 characters !
499               char[] cbuf = new char[4];
500               if (read(pr, cbuf) == 4) {
501                 c = Integer.parseInt(new String(cbuf), 16);
502                 buf.append((char)c);
503               }
504               break;
505 
506             // just an escaped character, unescape
507             default:
508               buf.append((char)c);
509           }
510           break;
511 
512         // eof
513         case -1: // fall through
514 
515           // separator token
516         case TOKEN_EQ:
517         case TOKEN_VAL_CLOS:
518           pr.unread(c);
519           return buf.toString();
520 
521         // no escaping
522         default:
523           buf.append((char)c);
524       }
525     }
526   }

529   private int nextToken(PushbackReader pr) throws IOException {
530     int c = ignorableWhiteSpace(pr);
531 
532     // immediately return EOF
533     if (c < 0) {
534       return (token = c);
535     }
536 
537     // check whether there is a name
538     if (NAME_CHARS.get(c) || !TOKEN_CHARS.get(c)) {
539       // read the property name
540       pr.unread(c);
541       tokenValue = readQuoted(pr);
542       return (token = TOKEN_NAME);
543     }
544 
545     // check another token
546     if (TOKEN_CHARS.get(c)) {
547       return (token = c);
548     }
549 
550     // unexpected character -> so what ??
551     return (token = -1);
552   }

555   private int ignorableWhiteSpace(PushbackReader pr) throws IOException {
556     int c = read(pr);
557     while (c >= 0 && Character.isWhitespace((char)c)) {
558       c = read(pr);
559     }
560     return c;
561   }

564   private int read(PushbackReader pr) throws IOException {
565     int c = pr.read();
566     if (c == '\r') {
567       int c1 = pr.read();
568       if (c1 != '\n') {
569         pr.unread(c1);
570       }
571       c = '\n';
572     }
573 
574     if (c == '\n') {
575       line++;
576       pos = 0;
577     }
578     else {
579       pos++;
580     }
581 
582     return c;
583   }

586   private int read(PushbackReader pr, char[] buf) throws IOException {
587     for (int i = 0; i < buf.length; i++) {
588       int c = read(pr);
589       if (c >= 0) {
590         buf[i] = (char)c;
591       }
592       else {
593         return i;
594       }
595     }
596 
597     return buf.length;
598   }

601   private IOException readFailure(int current, int expected) {
602     return new IOException("Unexpected token " + current + "; expected: " + expected + " (line=" + line + ", pos="
603         + pos + ")");
604   }

609   private static void writeValue(Writer out, Object value) throws IOException {
610     Class clazz = value.getClass();
611     if (clazz.isArray()) {
612       writeArray(out, value);
613     }
614     else if (value instanceof Collection) {
615       writeCollection(out, (Collection)value);
616     }
617     else {
618       writeType(out, clazz);
619       writeSimple(out, value);
620     }
621   }

624   private static void writeArray(Writer out, Object arrayValue) throws IOException {
625     int size = Array.getLength(arrayValue);
626     writeType(out, arrayValue.getClass().getComponentType());
627     out.write(TOKEN_ARR_OPEN);
628     for (int i = 0; i < size; i++) {
629       if (i > 0) {
630         out.write(TOKEN_COMMA);
631       }
632       writeSimple(out, Array.get(arrayValue, i));
633     }
634     out.write(TOKEN_ARR_CLOS);
635   }

637 
638   private static void writeCollection(Writer out, Collection collection) throws IOException {
639     if (collection.isEmpty()) {
640       out.write(TOKEN_VEC_OPEN);
641       out.write(TOKEN_VEC_CLOS);
642     }
643     else {
644       Iterator ci = collection.iterator();
645       Object firstElement = ci.next();
646 
647       writeType(out, firstElement.getClass());
648       out.write(TOKEN_VEC_OPEN);
649       writeSimple(out, firstElement);
650 
651       while (ci.hasNext()) {
652         out.write(TOKEN_COMMA);
653         writeSimple(out, ci.next());
654       }
655       out.write(TOKEN_VEC_CLOS);
656     }
657   }

660   private static void writeType(Writer out, Class valueType) throws IOException {
661     Integer code = (Integer)type2Code.get(valueType);
662     if (code != null) {
663       out.write((char)code.intValue());
664     }
665   }

668   private static void writeSimple(Writer out, Object value) throws IOException {
669     if (value instanceof Double) {
670       double dVal = ((Double)value).doubleValue();
671       value = new Long(Double.doubleToRawLongBits(dVal));
672     }
673     else if (value instanceof Float) {
674       float fVal = ((Float)value).floatValue();
675       value = new Integer(Float.floatToRawIntBits(fVal));
676     }
677 
678     out.write(TOKEN_VAL_OPEN);
679     writeQuoted(out, String.valueOf(value));
680     out.write(TOKEN_VAL_CLOS);
681   }

684   private static void writeQuoted(Writer out, String simple) throws IOException {
685     if (simple == null || simple.length() == 0) {
686       return;
687     }
688 
689     char c = 0;
690     int len = simple.length();
691     for (int i = 0; i < len; i++) {
692       c = simple.charAt(i);
693       switch (c) {
694         case '\\':
695         case TOKEN_VAL_CLOS:
696         case ' ':
697         case TOKEN_EQ:
698           out.write('\\');
699           out.write(c);
700           break;
701 
702         // well known escapes
703         case '\b':
704           out.write("\\b");
705           break;
706         case '\t':
707           out.write("\\t");
708           break;
709         case '\n':
710           out.write("\\n");
711           break;
712         case '\f':
713           out.write("\\f");
714           break;
715         case '\r':
716           out.write("\\r");
717           break;
718 
719         // other escaping
720         default:
721           if (c < ' ') {
722             String t = "000" + Integer.toHexString(c);
723             out.write("\\u" + t.substring(t.length() - 4));
724           }
725           else {
