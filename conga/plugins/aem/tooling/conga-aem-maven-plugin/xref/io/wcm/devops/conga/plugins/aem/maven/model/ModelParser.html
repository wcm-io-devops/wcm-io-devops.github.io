45  /**
46   * Parsers model.yaml files generated by CONGA.
47   */
48  public final class ModelParser {
49  
50    /**
51     * Model file.
52     */
53    public static final String MODEL_FILE = "model.yaml";
54  
55    private final Yaml yaml;
56  
57    /**
58     * Constructor
59     */
60    public ModelParser() {
61      this.yaml = YamlUtil.createYaml();
62    }
63  
64    /**
65     * Parses model.yaml file for given node and returns all content packages references in this fileData.
66     * @param nodeDir Node directory
67     * @return List of content packages
68     */
69    public List<ModelContentPackageFile> getContentPackagesForNode(File nodeDir) {
70      Map<String, Object> data = getModelData(nodeDir);
71      return collectPackages(data, nodeDir);
72    }
73  
74    /**
75     * Checks if the node has the given node role assigned.
76     * @param nodeDir Node directory
77     * @param roleName Node role name
78     * @return true if role is assigned
79     */
80    @SuppressWarnings("unchecked")
81    public boolean hasRole(File nodeDir, String roleName) {
82      Map<String, Object> data = getModelData(nodeDir);
83      List<Map<String, Object>> roles = (List<Map<String, Object>>)data.get("roles");
84      for (Map<String, Object> role : roles) {
85        if (StringUtils.equals(Objects.toString(role.get("role"), null), roleName)) {
86          return true;
87        }
88      }
89      return false;
90    }
91  
92    /**
93     * Collects all assigned "cloudManager.target" values (lists or single values) to any role from the node.
94     * @param nodeDir Node directory
95     * @return List of cloud manager environment names or "none"
96     */
97    @SuppressWarnings("unchecked")
98    public Set<String> getCloudManagerTarget(File nodeDir) {
99      Set<String> targets = new LinkedHashSet<>();
100     Map<String, Object> data = getModelData(nodeDir);
101     List<Map<String, Object>> roles = (List<Map<String, Object>>)data.get("roles");
102     for (Map<String, Object> role : roles) {
103       Map<String, Object> config = (Map<String, Object>)role.get("config");
104       if (config != null) {
105         Object targetValue = MapExpander.getDeep(config, "cloudManager.target");
106         if (targetValue != null) {
107           if (targetValue instanceof String) {
108             String target = (String)targetValue;
109             if (StringUtils.isNotBlank(target)) {
110               targets.add(target);
111             }
112           }
113           else if (targetValue instanceof List) {
114             targets.addAll(((List<String>)targetValue).stream()
115                 .filter(StringUtils::isNotBlank)
116                 .collect(Collectors.toList()));
117           }
118           else {
119             throw new RuntimeException("Invalid cloudManager.target value: " + targetValue);
120           }
121         }
122       }
123     }
124     return targets;
125   }
126 
127   private Map<String, Object> getModelData(File nodeDir) {
128     File modelFile = new File(nodeDir, MODEL_FILE);
129     if (!modelFile.exists() || !modelFile.isFile()) {
130       throw new RuntimeException("Model file not found: " + getCanonicalPath(modelFile));
131     }
132     return parseYaml(modelFile);
133   }
134 
135   @SuppressWarnings("unchecked")
136   private Map<String, Object> parseYaml(File modelFile) {
137     try {
138       try (InputStream is = new FileInputStream(modelFile);
139           Reader reader = new InputStreamReader(is, StandardCharsets.UTF_8)) {
140         return yaml.loadAs(reader, Map.class);
141       }
142     }
143     catch (IOException ex) {
144       throw new RuntimeException("Unable to parse " + getCanonicalPath(modelFile), ex);
145     }
146   }
147 
148   @SuppressWarnings("unchecked")
149   private List<ModelContentPackageFile> collectPackages(Map<String, Object> data, File nodeDir) {
150     List<ModelContentPackageFile> items = new ArrayList<>();
151     List<Map<String, Object>> roles = (List<Map<String, Object>>)data.get("roles");
152     if (roles != null) {
153       for (Map<String, Object> role : roles) {
154         List<Map<String, Object>> files = (List<Map<String, Object>>)role.get("files");
155         if (files != null) {
156           for (Map<String, Object> file : files) {
157             if (file.get(ContentPackagePropertiesPostProcessor.MODEL_OPTIONS_PROPERTY) != null) {
158               items.add(toContentPackageFile(file, role, nodeDir));
159             }
160           }
161         }
162       }
163     }
164     return items;
165   }
166 
167   private ModelContentPackageFile toContentPackageFile(Map<String, Object> fileData,
168       Map<String, Object> roleData, File nodeDir) {
169     String path = Objects.toString(fileData.get("path"), null);
170     File file = new File(nodeDir, path);
171     return new ModelContentPackageFile(file, fileData, roleData);
172   }
