68  /**
69   * Builds "all" package based on given set of content packages.
70   * <p>
71   * General concept:
72   * </p>
73   * <ul>
74   * <li>Iterates through all content packages that are generated or collected by CONGA and contained in the
75   * model.json</li>
76   * <li>Enforces the order defined in CONGA by automatically adding dependencies to all packages reflecting the file
77   * order in model.json</li>
78   * <li>Because the dependency chain may be different for each runmode (author/publish), each package is added once for
79   * each runmode (so usually twice, for author+publish)</li>
80   * <li>To avoid conflicts with duplicate packages with different dependencies all package names are changed and a
81   * runmode suffix (.author or .publish) is added, same applies to the install folder</li>
82   * <li>To avoid problems with nested sub packages, the sub packages are extracted from the packages and treated in the
83   * same way as other packages</li>
84   * </ul>
85   */
86  public final class AllPackageBuilder {

160   /**
161    * Add content packages and OSGi bundles to be contained in "all" content package.
162    * @param files Content packages (invalid will be filtered out) and OSGi bundles
163    * @param cloudManagerTarget Target environments/run modes the packages should be attached to
164    * @throws IllegalArgumentException If and invalid package type is detected
165    */
166   public void add(List<InstallableFile> files, Set<String> cloudManagerTarget) {
167     List<ContentPackageFile> contentPackages = filterFiles(files, ContentPackageFile.class);
168 
169     // collect list of cloud manager environment run modes
170     List<String> environmentRunModes = new ArrayList<>();
171     if (cloudManagerTarget.isEmpty()) {
172       environmentRunModes.add(RUNMODE_DEFAULT);
173     }
174     else {
175       environmentRunModes.addAll(cloudManagerTarget);
176     }
177 
178     List<ContentPackageFile> validContentPackages;
179     switch (packageTypeValidation) {
180       case STRICT:
181         validContentPackages = getValidContentPackagesStrictValidation(contentPackages);
182         break;
183       case WARN:
184         validContentPackages = getValidContentPackagesWarnValidation(contentPackages);
185         break;
186       default:
187         throw new IllegalArgumentException("Unsupported package type validation: " + packageTypeValidation);
188     }
189 
190     if (!validContentPackages.isEmpty()) {
191       contentPackageFileSets.add(new ContentPackageFileSet(validContentPackages, environmentRunModes));
192     }
193 
194     // add OSGi bundles
195     List<BundleFile> bundles = filterFiles(files, BundleFile.class);
196     if (!bundles.isEmpty()) {
197       bundleFileSets.add(new BundleFileSet(bundles, environmentRunModes));
198     }
199   }

201   /**
202    * Get valid content packages in strict mode: Ignore content packages without package type (with warning),
203    * fail build if Content package with "mixed" mode is found.
204    * @param contentPackages Content packages
205    * @return Valid content packages
206    */
207   private List<ContentPackageFile> getValidContentPackagesStrictValidation(List<? extends ContentPackageFile> contentPackages) {
208     // generate warning for each content packages without package type that is skipped
209     contentPackages.stream()
210         .filter(pkg -> !hasPackageType(pkg))
211         .forEach(pkg -> getLog().warn("Skipping content package without package type: {}" + getCanonicalPath(pkg.getFile())));
212 
213     // fail build if content packages with non-allowed package types exist
214     List<ContentPackageFile> invalidPackageTypeContentPackages = contentPackages.stream()
215         .filter(AllPackageBuilder::hasPackageType)
216         .filter(pkg -> !isValidPackageType(pkg))
217         .collect(Collectors.toList());
218     if (!invalidPackageTypeContentPackages.isEmpty()) {
219       throw new IllegalArgumentException("Content packages found with unsupported package types: " +
220           invalidPackageTypeContentPackages.stream()
221               .map(pkg -> pkg.getName() + " -> " + pkg.getPackageType())
222               .collect(Collectors.joining(", ")));
223     }
224 
225     // collect AEM content packages with package type
226     return contentPackages.stream()
227         .filter(AllPackageBuilder::hasPackageType)
228         .collect(Collectors.toList());
229   }

231   /**
232    * Get all content packages, generate warnings if package type is missing or "mixed" mode package type is used.
233    * @param contentPackages Content packages
234    * @return Valid content packages
235    */
236   private List<ContentPackageFile> getValidContentPackagesWarnValidation(List<? extends ContentPackageFile> contentPackages) {
237     // generate warning for each content packages without package type
238     contentPackages.stream()
239         .filter(pkg -> !hasPackageType(pkg))
240         .forEach(pkg -> getLog().warn("Found content package without package type: " + getCanonicalPath(pkg.getFile())));
241 
242     // generate warning for each content packages with invalid package type
243     contentPackages.stream()
244         .filter(AllPackageBuilder::hasPackageType)
245         .filter(pkg -> !isValidPackageType(pkg))
246         .forEach(pkg -> getLog().warn("Found content package with invalid package type: "
247             + getCanonicalPath(pkg.getFile()) + " -> " + pkg.getPackageType()));
248 
249     // return all content packages
250     return contentPackages.stream().collect(Collectors.toList());
251   }

253   private static <T> List<T> filterFiles(List<? extends InstallableFile> files, Class<T> fileClass) {
254     return files.stream()
255         .filter(fileClass::isInstance)
256         .map(fileClass::cast)
257         .collect(Collectors.toList());
258   }

260   /**
261    * Build "all" content package.
262    * @param properties Specifies additional properties to be set in the properties.xml file.
263    * @return true if "all" package was generated, false if no valid package was found.
264    * @throws IOException I/O exception
265    */
266   public boolean build(Map<String, String> properties) throws IOException {
267 
268     if (contentPackageFileSets.isEmpty()) {
269       return false;
270     }
271 
272     // prepare content package metadata
273     ContentPackageBuilder builder = new ContentPackageBuilder()
274         .group(groupName)
275         .name(packageName)
276         .packageType("container");
277     if (version != null) {
278       builder.version(version);
279     }
280 
281     // define root path for "all" package
282     String rootPath = buildRootPath(groupName, packageName);
283     builder.filter(new PackageFilter(rootPath));
284 
285     // additional package properties
286     if (properties != null) {
287       properties.entrySet().forEach(entry -> builder.property(entry.getKey(), entry.getValue()));
288     }
289 
290     // build set with dependencies instances for each package contained in all filesets
291     Set<Dependency> allPackagesFromFileSets = new HashSet<>();
292     for (ContentPackageFileSet fileSet : contentPackageFileSets) {
293       for (ContentPackageFile pkg : fileSet.getContentPackages()) {
294         addDependencyInformation(allPackagesFromFileSets, pkg);
295       }
296     }
297 
298     // build content package
299     // if auto dependencies is active: build separate "dependency chains" between mutable and immutable packages
300     try (ContentPackage contentPackage = builder.build(targetFile)) {
301       for (ContentPackageFileSet fileSet : contentPackageFileSets) {
302         for (String environmentRunMode : fileSet.getEnvironmentRunModes()) {
303           List<ContentPackageFile> previousPackages = new ArrayList<>();
304           for (ContentPackageFile pkg : fileSet.getContentPackages()) {
305 
306             ContentPackageFile previousPkg = null;
307 
308             if (autoDependenciesMode != AutoDependenciesMode.OFF
309                 && (autoDependenciesMode != AutoDependenciesMode.IMMUTABLE_ONLY || !isMutable(pkg))) {
310               // get last previous package
311               // if not IMMUTABLE_MUTABLE_COMBINED active only that of the same mutability type
312               previousPkg = previousPackages.stream()
313                   .filter(item -> (autoDependenciesMode == AutoDependenciesMode.IMMUTABLE_MUTABLE_COMBINED) || mutableMatches(item, pkg))
314                   .reduce((first, second) -> second)
315                   .orElse(null);
316             }
317 
318             // set package name, wire previous package in package dependency
319             List<TemporaryContentPackageFile> processedFiles = processContentPackage(pkg, previousPkg, environmentRunMode, allPackagesFromFileSets);
320 
321             // add processed content packages to "all" content package - and delete the temporary files
322             try {
323               for (TemporaryContentPackageFile processedFile : processedFiles) {
324                 String path = buildPackagePath(processedFile, rootPath, environmentRunMode);
325                 contentPackage.addFile(path, processedFile.getFile());
326                 if (log.isDebugEnabled()) {
327                   log.debug("  Add " + processedFile.getPackageInfoWithDependencies());
328                 }
329               }
330             }
331             finally {
332               processedFiles.stream()
333                   .map(TemporaryContentPackageFile::getFile)
334                   .forEach(FileUtils::deleteQuietly);
335             }
336 
337             previousPackages.add(pkg);
338           }
339         }
340       }
341       for (BundleFileSet bundleFileSet : bundleFileSets) {
342         for (String environmentRunMode : bundleFileSet.getEnvironmentRunModes()) {
343           for (BundleFile bundleFile : bundleFileSet.getBundles()) {
344             String path = buildBundlePath(bundleFile, rootPath, environmentRunMode);
345             contentPackage.addFile(path, bundleFile.getFile());
346           }
347         }
348       }
349     }
350 
351     return true;
352   }

354   private static boolean hasPackageType(ContentPackageFile pkg) {
355     // accept only content packages with a package type set
356     return pkg.getPackageType() != null;
357   }

359   private static boolean isValidPackageType(ContentPackageFile pkg) {
360     // check if the package type is an allowed package type
361     return ALLOWED_PACKAGE_TYPES.contains(pkg.getPackageType());
362   }
363 
364   private static boolean isMutable(ContentPackageFile pkg) {
365     return StringUtils.equals("content", pkg.getPackageType());
366   }
367 
368   private static boolean mutableMatches(ContentPackageFile pkg1, ContentPackageFile pkg2) {
369     if (pkg1 == null || pkg2 == null) {
370       return false;
371     }
372     return isMutable(pkg1) == isMutable(pkg2);
373   }

385   /**
386    * Generate suffix for instance and environment run modes.
387    * @param file Content package
388    * @return Package path
389    */
390   private static String buildRunModeSuffix(InstallableFile file, String environmentRunMode) {
391     StringBuilder runModeSuffix = new StringBuilder();
392     if (RunModeUtil.isOnlyAuthor(file)) {
393       runModeSuffix.append(".").append(RUNMODE_AUTHOR);
394     }
395     else if (RunModeUtil.isOnlyPublish(file)) {
396       runModeSuffix.append(".").append(RUNMODE_PUBLISH);
397     }
398     if (!StringUtils.equals(environmentRunMode, RUNMODE_DEFAULT)) {
399       runModeSuffix.append(".").append(environmentRunMode);
400     }
401     return runModeSuffix.toString();
402   }

404   /**
405    * Build path to be used for embedded package.
406    * @param pkg Package
407    * @param rootPath Root path
408    * @return Package path
409    */
410   private String buildPackagePath(ContentPackageFile pkg, String rootPath, String environmentRunMode) {
411     if (packageTypeValidation == PackageTypeValidation.STRICT && !isValidPackageType(pkg)) {
412       throw new IllegalArgumentException("Package " + pkg.getPackageInfo() + " has invalid package type: '" + pkg.getPackageType() + "'.");
413     }
414 
415     String runModeSuffix = buildRunModeSuffix(pkg, environmentRunMode);
416 
417     // add run mode suffix to both install folder path and package file name
418     String path = rootPath + "/" + StringUtils.defaultString(pkg.getPackageType(), "misc") + "/install" + runModeSuffix;
419 
420     String versionSuffix = "";
421     String packageVersion = pkg.getVersion();
422     if (packageVersion != null && pkg.getFile().getName().contains(packageVersion)) {
423       versionSuffix = "-" + packageVersion;
424     }
425     String fileName = pkg.getName() + versionSuffix
426         + "." + FilenameUtils.getExtension(pkg.getFile().getName());
427     return path + "/" + fileName;
428   }

430   /**
431    * Build path to be used for embedded bundle.
432    * @param bundleFile Bundle
433    * @param rootPath Root path
434    * @return Package path
435    */
436   private static String buildBundlePath(BundleFile bundleFile, String rootPath, String environmentRunMode) {
437     String runModeSuffix = buildRunModeSuffix(bundleFile, environmentRunMode);
438 
439     // add run mode suffix to both install folder path and package file name
440     String path = rootPath + "/application/install" + runModeSuffix;
441 
442     return path + "/" + bundleFile.getFile().getName();
443   }

445   /**
446    * Rewrite content package ZIP file while adding to "all" package:
447    * Add dependency to previous package in CONGA configuration file oder.
448    * @param pkg Package to process (can be parent packe of the actual file)
449    * @param previousPkg Previous package to get dependency information from.
450    *          Is null if no previous package exists or auto dependency mode is switched off.
451    * @param environmentRunMode Environment run mode
452    * @param allPackagesFromFileSets Set with all packages from all file sets as dependency instances
453    * @return Returns a list of content package *temporary* files - have to be deleted when processing is completed.
454    * @throws IOException I/O error
455    */
456   private List<TemporaryContentPackageFile> processContentPackage(ContentPackageFile pkg,
457       ContentPackageFile previousPkg, String environmentRunMode,
458       Set<Dependency> allPackagesFromFileSets) throws IOException {
459 
460     List<TemporaryContentPackageFile> result = new ArrayList<>();
461     List<TemporaryContentPackageFile> subPackages = new ArrayList<>();
462 
463     // create temp zip file to create rewritten copy of package
464     File tempFile = File.createTempFile(FilenameUtils.getBaseName(pkg.getFile().getName()), ".zip");
465 
466     // open original content package
467     try (ZipFile zipFileIn = new ZipFile(pkg.getFile())) {
468 
469       // iterate through entries and write them to the temp. zip file
470       try (FileOutputStream fos = new FileOutputStream(tempFile);
471           ZipOutputStream zipOut = new ZipOutputStream(fos)) {
472         Enumeration<? extends ZipEntry> zipInEntries = zipFileIn.entries();
473         while (zipInEntries.hasMoreElements()) {
474           ZipEntry zipInEntry = zipInEntries.nextElement();
475           if (!zipInEntry.isDirectory()) {
476             try (InputStream is = zipFileIn.getInputStream(zipInEntry)) {
477               boolean processedEntry = false;
478 
479               // if entry is properties.xml, update dependency information
480               if (StringUtils.equals(zipInEntry.getName(), "META-INF/vault/properties.xml")) {
481                 Properties props = new Properties();
482                 props.loadFromXML(is);
483                 addSuffixToPackageName(props, pkg, environmentRunMode);
484 
485                 // update package dependencies
486                 ContentPackageFile dependencyFile = previousPkg;
487                 if (autoDependenciesMode == AutoDependenciesMode.OFF) {
488                   dependencyFile = null;
489                 }
490                 updateDependencies(props, dependencyFile, environmentRunMode, allPackagesFromFileSets);
491 
492                 // if package type is missing package properties, put in the type defined in model
493                 String packageType = pkg.getPackageType();
494                 if (props.get(NAME_PACKAGE_TYPE) == null && packageType != null) {
495                   props.put(NAME_PACKAGE_TYPE, packageType);
496                 }
497 
498                 ZipEntry zipOutEntry = new ZipEntry(zipInEntry.getName());
499                 zipOut.putNextEntry(zipOutEntry);
500                 props.storeToXML(zipOut, null);
501                 processedEntry = true;
502               }
503 
504               // process sub-packages as well: add runmode suffix and update dependencies
505               else if (StringUtils.equals(FilenameUtils.getExtension(zipInEntry.getName()), "zip")) {
506                 File tempSubPackageFile = File.createTempFile(FilenameUtils.getBaseName(zipInEntry.getName()), ".zip");
507                 try (FileOutputStream subPackageFos = new FileOutputStream(tempSubPackageFile)) {
508                   IOUtils.copy(is, subPackageFos);
509                 }
510 
511                 // check if contained ZIP file is really a content package
512                 // then process it as well, remove if from the content package is was contained it
513                 // and add it as "1st level package" to the all package
514                 TemporaryContentPackageFile tempSubPackage = new TemporaryContentPackageFile(tempSubPackageFile, pkg.getVariants());
515                 if (packageTypeValidation == PackageTypeValidation.STRICT && !isValidPackageType(tempSubPackage)) {
516                   throw new IllegalArgumentException("Package " + pkg.getPackageInfo() + " contains sub package " + tempSubPackage.getPackageInfo()
517                       + " with invalid package type: '" + StringUtils.defaultString(tempSubPackage.getPackageType()) + "'");
518                 }
519                 if (StringUtils.isNoneBlank(tempSubPackage.getGroup(), tempSubPackage.getName())) {
520                   subPackages.add(tempSubPackage);
521                   processedEntry = true;
522                 }
523                 else {
524                   FileUtils.deleteQuietly(tempSubPackageFile);
525                 }
526               }
527 
528               // otherwise transfer the binary data 1:1
529               if (!processedEntry) {
530                 ZipEntry zipOutEntry = new ZipEntry(zipInEntry.getName());
531                 zipOut.putNextEntry(zipOutEntry);
532                 IOUtils.copy(is, zipOut);
533               }
534             }
535 
536             zipOut.closeEntry();
537           }
538         }
539       }
540 
541       // add sub package metadata to set with dependency information
542       for (TemporaryContentPackageFile tempSubPackage : subPackages) {
543         addDependencyInformation(allPackagesFromFileSets, tempSubPackage);
544       }
545 
546       // process sub packages and add to result
547       for (TemporaryContentPackageFile tempSubPackage : subPackages) {
548         result.addAll(processContentPackage(tempSubPackage, previousPkg, environmentRunMode, allPackagesFromFileSets));
549       }
550 
551       result.add(new TemporaryContentPackageFile(tempFile, pkg.getVariants()));
552     }
553     return result;
554   }

556   /**
557    * Add dependency information to dependencies string in properties (if it does not exist already).
558    * @param props Properties
559    * @param dependencyFile Dependency package
560    * @param allPackagesFromFileSets Set with all packages from all file sets as dependency instances
561    */
562   private static void updateDependencies(Properties props, ContentPackageFile dependencyFile, String environmentRunMode,
563       Set<Dependency> allPackagesFromFileSets) {
564     String[] existingDepsStrings = StringUtils.split(props.getProperty(NAME_DEPENDENCIES), ",");
565     Dependency[] existingDeps = null;
566     if (existingDepsStrings != null && existingDepsStrings.length > 0) {
567       existingDeps = Dependency.fromString(existingDepsStrings);
568     }
569     if (existingDeps != null) {
570       existingDeps = removeReferencesToManagedPackages(existingDeps, allPackagesFromFileSets);
571     }
572 
573     Dependency[] deps;
574     if (dependencyFile != null) {
575       String runModeSuffix = buildRunModeSuffix(dependencyFile, environmentRunMode);
576       Dependency newDependency = new Dependency(dependencyFile.getGroup(),
577           dependencyFile.getName() + runModeSuffix,
578           VersionRange.fromString(dependencyFile.getVersion()));
579       deps = addDependency(existingDeps, newDependency);
580     }
581     else {
582       deps = existingDeps;
583     }
584 
585     if (deps != null) {
586       String dependenciesString = Dependency.toString(deps);
587       props.put(NAME_DEPENDENCIES, dependenciesString);
588     }
589   }
590 
591   private static Dependency[] addDependency(Dependency[] existingDeps, Dependency newDependency) {
592     if (existingDeps != null) {
593       return DependencyUtil.add(existingDeps, newDependency);
594     }
595     else {
596       return new Dependency[] { newDependency };
597     }
598   }

600   private static void addSuffixToPackageName(Properties props, ContentPackageFile pkg, String environmentRunMode) {
601     String runModeSuffix = buildRunModeSuffix(pkg, environmentRunMode);
602     String packageName = props.getProperty(NAME_NAME) + runModeSuffix;
603     props.put(NAME_NAME, packageName);
604   }

606   /**
607    * Removes existing references to packages contained in the list of packages to manage by this builder because
608    * they are added new (and probably with a different package name) during processing.
609    * @param deps Dependencies list
610    * @param allPackagesFromFileSets Set with all packages from all file sets as dependency instances
611    * @return Dependencies list
612    */
613   private static Dependency[] removeReferencesToManagedPackages(Dependency[] deps, Set<Dependency> allPackagesFromFileSets) {
614     return Arrays.stream(deps)
615         .filter(dep -> !allPackagesFromFileSets.contains(dep))
616         .toArray(size -> new Dependency[size]);
617   }

619   private static void addDependencyInformation(Set<Dependency> allPackagesFromFileSets, ContentPackageFile pkg) {
620     allPackagesFromFileSets.add(new Dependency(pkg.getGroup(), pkg.getName(), VersionRange.fromString(pkg.getVersion())));
621   }
