131   /**
132    * Add content packages to be contained in "all" content package.
133    * @param contentPackages Content packages (invalid will be filtered out)
134    * @param cloudManagerTarget Target environments/run modes the packages should be attached to
135    * @throws IllegalArgumentException If and invalid package type is detected
136    */
137   public void add(List<ContentPackageFile> contentPackages, Set<String> cloudManagerTarget) {
138 
139     // collect list of cloud manager environment run modes
140     List<String> environmentRunModes = new ArrayList<>();
141     if (cloudManagerTarget.isEmpty()) {
142       environmentRunModes.add(RUNMODE_DEFAULT);
143     }
144     else {
145       environmentRunModes.addAll(cloudManagerTarget);
146     }
147 
148     // generate warnings for each invalid content packages that is skipped
149     contentPackages.stream()
150         .filter(pkg -> !hasPackageType(pkg))
151         .forEach(pkg -> getLog().warn("Skipping content package without package type: " + getCanonicalPath(pkg.getFile())));
152 
153     // fail build if content packages with non-allowed package types exist
154     List<ContentPackageFile> invalidPackageTypeContentPackages = contentPackages.stream()
155         .filter(AllPackageBuilder::hasPackageType)
156         .filter(pkg -> !isValidPackageType(pkg))
157         .collect(Collectors.toList());
158     if (!invalidPackageTypeContentPackages.isEmpty()) {
159       throw new IllegalArgumentException("Content packages found with unsupported package types: " +
160           invalidPackageTypeContentPackages.stream()
161               .map(pkg -> pkg.getName() + " -> " + pkg.getPackageType())
162               .collect(Collectors.joining(", ")));
163     }
164 
165     // collect AEM content packages for this node
166     List<ContentPackageFile> validContentPackages = contentPackages.stream()
167         .filter(AllPackageBuilder::hasPackageType)
168         .collect(Collectors.toList());
169 
170     if (!validContentPackages.isEmpty()) {
171       fileSets.add(new ContentPackageFileSet(validContentPackages, environmentRunModes));
172     }
173   }

175   /**
176    * Build "all" content package.
177    * @param properties Specifies additional properties to be set in the properties.xml file.
178    * @return true if "all" package was generated, false if no valid package was found.
179    * @throws IOException I/O exception
180    */
181   public boolean build(Map<String, String> properties) throws IOException {
182 
183     if (fileSets.isEmpty()) {
184       return false;
185     }
186 
187     // prepare content package metadata
188     ContentPackageBuilder builder = new ContentPackageBuilder()
189         .group(groupName)
190         .name(packageName)
191         .packageType("container");
192     if (version != null) {
193       builder.version(version);
194     }
195 
196     // define root path for "all" package
197     String rootPath = buildRootPath(groupName, packageName);
198     builder.filter(new PackageFilter(rootPath));
199 
200     // additional package properties
201     if (properties != null) {
202       properties.entrySet().forEach(entry -> builder.property(entry.getKey(), entry.getValue()));
203     }
204 
205     // build set with dependencies instances for each package contained in all filesets
206     Set<Dependency> allPackagesFromFileSets = new HashSet<>();
207     for (ContentPackageFileSet fileSet : fileSets) {
208       for (ContentPackageFile pkg : fileSet.getContentPackages()) {
209         allPackagesFromFileSets.add(new Dependency(pkg.getGroup(), pkg.getName(), VersionRange.fromString(pkg.getVersion())));
210       }
211     }
212 
213     // build content package
214     // if auto dependencies is active: build separate "dependency chains" between mutable and immutable packages
215     try (ContentPackage contentPackage = builder.build(targetFile)) {
216       for (ContentPackageFileSet fileSet : fileSets) {
217         for (String environmentRunMode : fileSet.getEnvironmentRunModes()) {
218           List<ContentPackageFile> previousPackages = new ArrayList<>();
219           for (ContentPackageFile pkg : fileSet.getContentPackages()) {
220             String path = buildPackagePath(pkg, rootPath, environmentRunMode);
221 
222             ContentPackageFile previousPkg = null;
223 
224             if (autoDependenciesMode != AutoDependenciesMode.OFF
225                 && (autoDependenciesMode != AutoDependenciesMode.IMMUTABLE_ONLY || !isMutable(pkg))) {
226               // get last previous package
227               // if not IMMUTABLE_MUTABLE_COMBINED active only that of the same mutability type
228               previousPkg = previousPackages.stream()
229                   .filter(item -> (autoDependenciesMode == AutoDependenciesMode.IMMUTABLE_MUTABLE_COMBINED) || mutableMatches(item, pkg))
230                   .reduce((first, second) -> second)
231                   .orElse(null);
232             }
233 
234             // set package name, wire previous package in package dependency
235             File tempPackageFile = processContentPackage(pkg.getFile(), pkg, previousPkg, environmentRunMode, allPackagesFromFileSets);
236 
237             // add temp zip file to "all" content package
238             try {
239               contentPackage.addFile(path, tempPackageFile);
240             }
241             finally {
242               FileUtils.deleteQuietly(tempPackageFile);
243             }
244 
245             previousPackages.add(pkg);
246           }
247         }
248       }
249     }
250 
251     return true;
252   }

254   private static boolean hasPackageType(ContentPackageFile pkg) {
255     // accept only content packages with a package type set
256     return pkg.getPackageType() != null;
257   }
258 
259   private static boolean isValidPackageType(ContentPackageFile pkg) {
260     // check if the package type is an allowed package type
261     return ALLOWED_PACKAGE_TYPES.contains(pkg.getPackageType());
262   }
263 
264   private static boolean isMutable(ContentPackageFile pkg) {
265     return StringUtils.equals("content", pkg.getPackageType());
266   }
267 
268   private static boolean mutableMatches(ContentPackageFile pkg1, ContentPackageFile pkg2) {
269     if (pkg1 == null || pkg2 == null) {
270       return false;
271     }
272     return isMutable(pkg1) == isMutable(pkg2);
273   }

275   /**
276    * Build root path to be used for embedded package.
277    * @param groupName Group name
278    * @param packageName Package name
279    * @return Package path
280    */
281   private static String buildRootPath(String groupName, String packageName) {
282     return "/apps/" + groupName + "-" + packageName + "-packages";
283   }

285   /**
286    * Generate suffix for instance and environment run modes.
287    * @param pkg Package
288    * @return Package path
289    */
290   private static String buildRunModeSuffix(ContentPackageFile pkg, String environmentRunMode) {
291     StringBuilder runModeSuffix = new StringBuilder();
292     if (RunModeUtil.isOnlyAuthor(pkg)) {
293       runModeSuffix.append(".").append(RUNMODE_AUTHOR);
294     }
295     else if (RunModeUtil.isOnlyPublish(pkg)) {
296       runModeSuffix.append(".").append(RUNMODE_PUBLISH);
297     }
298     if (!StringUtils.equals(environmentRunMode, RUNMODE_DEFAULT)) {
299       runModeSuffix.append(".").append(environmentRunMode);
300     }
301     return runModeSuffix.toString();
302   }

304   /**
305    * Build path to be used for embedded package.
306    * @param pkg Package
307    * @param rootPath Root path
308    * @return Package path
309    */
310   private static String buildPackagePath(ContentPackageFile pkg, String rootPath, String environmentRunMode) {
311     String runModeSuffix = buildRunModeSuffix(pkg, environmentRunMode);
312 
313     // add run mode suffix to both install folder path and package file name
314     String path = rootPath + "/" + pkg.getPackageType() + "/install" + runModeSuffix;
315 
316     String versionSuffix = "";
317     if (pkg.getVersion() != null && pkg.getFile().getName().contains(pkg.getVersion())) {
318       versionSuffix = "-" + pkg.getVersion();
319     }
320     String fileName = pkg.getName() + runModeSuffix + versionSuffix
321         + "." + FilenameUtils.getExtension(pkg.getFile().getName());
322     return path + "/" + fileName;
323   }

325   /**
326    * Rewrite content package ZIP file while adding to "all" package:
327    * Add dependency to previous package in CONGA configuration file oder.
328    * @param contentPackageFile The actual content package file to process
329    * @param pkg Package to process (can be parent packe of the actual file)
330    * @param previousPkg Previous package to get dependency information from.
331    *          Is null if no previous package exists or auto dependency mode is switched off.
332    * @param environmentRunMode Environment run mode
333    * @param allPackagesFromFileSets Set with all packages from all file sets as dependency instances
334    * @return Returns a *temporary* file - has to be deleted when processing the result is completed.
335    * @throws IOException I/O error
336    */
337   private File processContentPackage(File contentPackageFile, ContentPackageFile pkg,
338       ContentPackageFile previousPkg, String environmentRunMode,
339       Set<Dependency> allPackagesFromFileSets) throws IOException {
340 
341     // create temp zip file to create rewritten copy of package
342     File tempFile = File.createTempFile("pkg", ".zip");
343 
344     // open original content package
345     try (ZipFile zipFileIn = new ZipFile(contentPackageFile)) {
346       String dependenciesString = null;
347 
348       // iterate through entries and write them to the temp. zip file
349       try (FileOutputStream fos = new FileOutputStream(tempFile);
350           ZipOutputStream zipOut = new ZipOutputStream(fos)) {
351         Enumeration<? extends ZipEntry> zipInEntries = zipFileIn.entries();
352         while (zipInEntries.hasMoreElements()) {
353           ZipEntry zipInEntry = zipInEntries.nextElement();
354           if (!zipInEntry.isDirectory()) {
355             try (InputStream is = zipFileIn.getInputStream(zipInEntry)) {
356 
357               // if entry is properties.xml, update dependency information
358               if (StringUtils.equals(zipInEntry.getName(), "META-INF/vault/properties.xml")) {
359                 Properties props = new Properties();
360                 props.loadFromXML(is);
361                 addSuffixToPackageName(props, pkg, environmentRunMode);
362                 if (autoDependenciesMode != AutoDependenciesMode.OFF) {
363                   dependenciesString = updateDependencies(props, previousPkg, environmentRunMode, allPackagesFromFileSets);
364                 }
365 
366                 ZipEntry zipOutEntry = new ZipEntry(zipInEntry.getName());
367                 zipOut.putNextEntry(zipOutEntry);
368                 props.storeToXML(zipOut, null);
369               }
370 
371               // process sub-packages as well: add runmode suffix and update dependencies
372               else if (StringUtils.equals(FilenameUtils.getExtension(zipInEntry.getName()), "zip")) {
373                 String path = FilenameUtils.getPath(zipInEntry.getName());
374                 String basename = FilenameUtils.getBaseName(zipInEntry.getName());
375                 String runModeSuffix = buildRunModeSuffix(pkg, environmentRunMode);
376 
377                 File tempSubPackageFile = File.createTempFile("subpkg-" + basename + runModeSuffix, ".zip");
378                 try (FileOutputStream subPackageFos = new FileOutputStream(tempSubPackageFile)) {
379                   IOUtils.copy(is, subPackageFos);
380                 }
381                 File resultSubPackageFile = processContentPackage(tempSubPackageFile, pkg, null, environmentRunMode, allPackagesFromFileSets);
382                 try (FileInputStream subPackageFis = new FileInputStream(resultSubPackageFile)) {
383                   ZipEntry zipOutEntry = new ZipEntry(path + basename + runModeSuffix + ".zip");
384                   zipOut.putNextEntry(zipOutEntry);
385                   IOUtils.copy(subPackageFis, zipOut);
386                 }
387                 finally {
388                   FileUtils.deleteQuietly(resultSubPackageFile);
389                 }
390               }
391 
392               // otherwise transfer the binary data 1:1
393               else {
394                 ZipEntry zipOutEntry = new ZipEntry(zipInEntry.getName());
395                 zipOut.putNextEntry(zipOutEntry);
396                 IOUtils.copy(is, zipOut);
397               }
398             }
399 
400             zipOut.closeEntry();
401           }
402         }
403 
404         if (log.isDebugEnabled()) {
405           log.debug("Processed " + getCanonicalPath(contentPackageFile)
406               + (dependenciesString != null ? " with dependencies: " + dependenciesString : ""));
407         }
408       }
409 
410       return tempFile;
411     }
412   }

414   /**
415    * Add dependency information to dependencies string in properties (if it does not exist already).
416    * @param props Properties
417    * @param dependencyFile Dependency package
418    * @param allPackagesFromFileSets Set with all packages from all file sets as dependency instances
419    * @throws IOException I/O exception
420    */
421   private static String updateDependencies(Properties props, ContentPackageFile dependencyFile, String environmentRunMode,
422       Set<Dependency> allPackagesFromFileSets) throws IOException {
423     String[] existingDepsStrings = StringUtils.split(props.getProperty(NAME_DEPENDENCIES), ",");
424     Dependency[] existingDeps = null;
425     if (existingDepsStrings != null && existingDepsStrings.length > 0) {
426       existingDeps = Dependency.fromString(existingDepsStrings);
427     }
428     if (existingDeps != null) {
429       existingDeps = removeReferencesToManagedPackages(existingDeps, allPackagesFromFileSets);
430     }
431 
432     Dependency[] deps;
433     if (dependencyFile != null) {
434       String runModeSuffix = buildRunModeSuffix(dependencyFile, environmentRunMode);
435       Dependency newDependency = new Dependency(dependencyFile.getGroup(),
436           dependencyFile.getName() + runModeSuffix,
437           VersionRange.fromString(dependencyFile.getVersion()));
438       deps = addDependency(existingDeps, newDependency);
439     }
440     else {
441       deps = existingDeps;
442     }
443 
444     if (deps != null) {
445       String dependenciesString = Dependency.toString(deps);
446       props.put(NAME_DEPENDENCIES, dependenciesString);
447       return dependenciesString;
448     }
449     else {
450       return null;
451     }
452   }
453 
454   private static Dependency[] addDependency(Dependency[] existingDeps, Dependency newDependency) {
455     if (existingDeps != null) {
456       return DependencyUtil.add(existingDeps, newDependency);
457     }
458     else {
459       return new Dependency[] { newDependency };
460     }
461   }
462 
463   private static void addSuffixToPackageName(Properties props, ContentPackageFile pkg, String environmentRunMode) {
464     String runModeSuffix = buildRunModeSuffix(pkg, environmentRunMode);
465     String packageName = props.getProperty(NAME_NAME) + runModeSuffix;
466     props.put(NAME_NAME, packageName);
467   }

469   /**
470    * Removes existing references to packages contained in the list of packages to manage by this builder because
471    * they are added new (and probably with a different package name) during processing.
472    * @param deps Dependencies list
473    * @param allPackagesFromFileSets Set with all packages from all file sets as dependency instances
474    * @return Dependencies list
475    */
476   private static Dependency[] removeReferencesToManagedPackages(Dependency[] deps, Set<Dependency> allPackagesFromFileSets) {
477     return Arrays.stream(deps)
478         .filter(dep -> !allPackagesFromFileSets.contains(dep))
479         .toArray(size -> new Dependency[size]);
480   }
