22  import static io.wcm.devops.conga.plugins.aem.postprocessor.ContentPackageOptions.PROPERTY_DEPENCY_CHAIN_IGNORE;
23  import static io.wcm.devops.conga.plugins.aem.postprocessor.ContentPackageOptions.PROPERTY_PACKAGE_PACKAGE_TYPE;
24  import static org.apache.jackrabbit.vault.packaging.PackageProperties.NAME_PACKAGE_TYPE;

26  import java.io.IOException;
27  import java.util.List;
28  import java.util.Map;
29  
30  import org.apache.commons.lang3.StringUtils;
31  import org.slf4j.Logger;
32  
33  import io.wcm.devops.conga.generator.GeneratorException;
34  import io.wcm.devops.conga.generator.plugins.postprocessor.AbstractPostProcessor;
35  import io.wcm.devops.conga.generator.spi.ImplicitApplyOptions;
36  import io.wcm.devops.conga.generator.spi.context.FileContext;
37  import io.wcm.devops.conga.generator.spi.context.PostProcessorContext;
38  import io.wcm.devops.conga.generator.util.FileUtil;
39  import io.wcm.devops.conga.plugins.aem.util.ContentPackageUtil;
40  import io.wcm.tooling.commons.packmgr.util.ContentPackageProperties;

42  /**
43   * Extracts properties from a given AEM content package and stores them as additional
44   * model options in the file context.
45   */
46  public class ContentPackagePropertiesPostProcessor extends AbstractPostProcessor {
47  
48    /**
49     * Plugin name
50     */
51    public static final String NAME = "aem-contentpackage-properties";
52  
53    /**
54     * Holds map containing the content package properties.
55     */
56    public static final String MODEL_OPTIONS_PROPERTY = "aemContentPackageProperties";
57  
58    /**
59     * Marks content package to be ignored in dependency chain.
60     */
61    public static final String DEPENDENCY_CHAIN_IGNORE_PROPERTY = "aemContentPackageDepenencyChainIgnore";
62  
63    private static final String FILE_EXTENSION = "zip";
64    private static final String ALTERNATE_FILE_EXTENSION = "jar";
65  
66    @Override
67    public String getName() {
68      return NAME;
69    }
70  
71    @Override
72    public boolean accepts(FileContext file, PostProcessorContext context) {
73      return FileUtil.matchesExtension(file, FILE_EXTENSION)
74          || FileUtil.matchesExtension(file, ALTERNATE_FILE_EXTENSION);
75    }
76  
77    @Override
78    public ImplicitApplyOptions implicitApply(FileContext file, PostProcessorContext context) {
79      if (FileUtil.matchesExtension(file, FILE_EXTENSION)) {
80        return ImplicitApplyOptions.ALWAYS;
81      }
82      else {
83        return ImplicitApplyOptions.NEVER;
84      }
85    }
86  
87    @Override
88    public List<FileContext> apply(FileContext fileContext, PostProcessorContext context) {
89      Logger logger = context.getLogger();
90  
91      try {
92        Map<String, Object> properties = ContentPackageProperties.get(fileContext.getFile());
93        if (!properties.isEmpty()) {
94  
95          // allow to redefine the packageType content package property via post processor options
96          // this is useful when 3rdparty packages do not define a package type
97          String packageType = ContentPackageUtil.getOptionalProp(context.getOptions(), PROPERTY_PACKAGE_PACKAGE_TYPE);
98          if (StringUtils.isNotBlank(packageType)) {
99            properties.put(NAME_PACKAGE_TYPE, packageType);
100         }
101 
102         fileContext.getModelOptions().put(MODEL_OPTIONS_PROPERTY, properties);
103 
104         boolean dependencyChainIgnore = ContentPackageUtil.getOptionalPropBoolean(context.getOptions(), PROPERTY_DEPENCY_CHAIN_IGNORE);
105         if (dependencyChainIgnore) {
106           fileContext.getModelOptions().put(DEPENDENCY_CHAIN_IGNORE_PROPERTY, true);
107         }
108 
109         logger.info("Extracted properties from AEM content package.");
110       }
111     }
112     catch (IOException ex) {
113       throw new GeneratorException("Unable to extract properties from AEM package " + fileContext.getCanonicalPath(), ex);
114     }
115 
116     return List.of();
117   }
118 
119 }
