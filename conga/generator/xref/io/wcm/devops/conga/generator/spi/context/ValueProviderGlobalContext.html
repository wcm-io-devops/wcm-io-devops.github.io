20  package io.wcm.devops.conga.generator.spi.context;
21  
22  import java.util.Collections;
23  import java.util.HashMap;
24  import java.util.Map;
25  
26  /**
27   * Global context for all {@link io.wcm.devops.conga.generator.spi.ValueProviderPlugin} implementations.
28   */
29  public final class ValueProviderGlobalContext extends AbstractPluginContext<ValueProviderGlobalContext> {
30  
31    private final Map<String, Object> globalValueProviderCache = new HashMap<>();
32  
33    /**
34     * Parameter name in value provider configuration to specify the value provider plugin name.
35     * If not defined, the value provider name itself is used as plugin name.
36     */
37    public static final String PARAM_PLUGIN_NAME = "_plugin_";
38  
39    /**
40     * @return Configuration for value providers.
41     *         The outer map uses the value provider plugin name as key, the inner map contain the config properties.
42     *         Never null.
43     */
44    public Map<String, Map<String, Object>> getValueProviderConfig() {
45      Map<String, Map<String, Object>> value = this.getPluginContextOptions().getValueProviderConfig();
46      if (value == null) {
47        value = Collections.emptyMap();
48      }
49      return value;
50    }
51  
52    /**
53     * @param valueProviderName Value provider name
54     * @return Configuration for the given value provider. Never null.
55     */
56    public Map<String, Object> getValueProviderConfig(String valueProviderName) {
57      Map<String, Object> config = getValueProviderConfig().get(valueProviderName);
58      if (config == null) {
59        return Collections.emptyMap();
60      }
61      return config;
62    }
63  
64    /**
65     * Get global cache map used by all value provider plugin implementations.
66     * @return Global cache map
67     */
68    public Map<String, Object> getGlobalValueProviderCache() {
69      return globalValueProviderCache;
70    }
71  
72  }
