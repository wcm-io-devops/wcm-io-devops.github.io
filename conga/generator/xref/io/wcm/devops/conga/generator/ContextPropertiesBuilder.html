53  /**
54   * Builds context variables
55   */
56  public final class ContextPropertiesBuilder {
57  
58    static final Map<String, Object> EMPTY_CONTEXT_VARIABLES = ImmutableMap.<String, Object>builder()
59        .put(VERSION, "")
60        .put(ENVIRONMENT, "")
61        .put(NODES, Collections.emptyList())
62        .put(NODES_BY_ROLE, Collections.emptyMap())
63        .put(NODES_BY_ROLE_VARIANT, Collections.emptyMap())
64        .put(TENANTS, Collections.emptyMap())
65        .put(TENANTS_BY_ROLE, Collections.emptyMap())
66        .put(ROLE, "")
67        .put(ROLE_VARIANT, "")
68        .put(NODE, "")
69        .put(TENANT, "")
70        .put(TENANT_ROLES, Collections.emptyList())
71        .build();
72  
73    private ContextPropertiesBuilder() {
74      // static methods only
75    }
76  
77    /**
78     * Build context variables that are global to the environment.
79     * @param environmentName Environment name
80     * @param environment Environment
81     * @param version Environment version
82     * @param variableObjectTreeResolver Variable object tree resolver
83     * @param variableStringResolver Variable string resolver
84     * @return Context variables map
85     */
86    public static Map<String, Object> buildEnvironmentContextVariables(String environmentName,
87        Environment environment, String version,
88        VariableObjectTreeResolver variableObjectTreeResolver, VariableStringResolver variableStringResolver) {
89      Map<String, Object> map = new HashMap<>(EMPTY_CONTEXT_VARIABLES);
90  
91      if (version != null) {
92        map.put(VERSION, version);
93      }
94      map.put(ENVIRONMENT, environmentName);
95  
96      // clone environment before resolving variables to make sure they are resolved only for this context, not for file generation
97      Environment clonedEnvironemnt = ObjectCloner.deepClone(environment);
98  
99      // resolve all variables at any level in environment
100     variableObjectTreeResolver.resolve(clonedEnvironemnt);
101 
102     // list of nodes
103     map.put(NODES, clonedEnvironemnt.getNodes());
104     Map<String, List<Node>> nodesByRole = new HashMap<>();
105     Map<String, Map<String, List<Node>>> nodesByRoleVariant = new HashMap<>();
106     for (Node node : clonedEnvironemnt.getNodes()) {
107       for (NodeRole nodeRole : node.getRoles()) {
108         List<Node> nodes = nodesByRole.get(nodeRole.getRole());
109         if (nodes == null) {
110           nodes = new ArrayList<>();
111           nodesByRole.put(nodeRole.getRole(), nodes);
112         }
113         nodes.add(node);
114 
115         List<String> variants = nodeRole.getAggregatedVariants();
116         for (String variant : variants) {
117           Map<String, List<Node>> nodesByVariant = nodesByRoleVariant.get(nodeRole.getRole());
118           if (nodesByVariant == null) {
119             nodesByVariant = new HashMap<>();
120             nodesByRoleVariant.put(nodeRole.getRole(), nodesByVariant);
121           }
122           List<Node> variantNodes = nodesByVariant.get(variant);
123           if (variantNodes == null) {
124             variantNodes = new ArrayList<>();
125             nodesByVariant.put(variant, variantNodes);
126           }
127           variantNodes.add(node);
128         }
129       }
130     }
131     map.put(NODES_BY_ROLE, nodesByRole);
132     map.put(NODES_BY_ROLE_VARIANT, nodesByRoleVariant);
133 
134     // list of tenants
135     map.put(TENANTS, clonedEnvironemnt.getTenants());
136     Map<String, List<Tenant>> tenantsByRole = new HashMap<>();
137     for (Tenant tenant : clonedEnvironemnt.getTenants()) {
138 
139       // resolve placeholders in tentant name
140       Map<String, Object> tenantConfig = MapMerger.merge(clonedEnvironemnt.getConfig(), tenant.getConfig());
141       String tenantName = variableStringResolver.resolveString(tenant.getTenant(), tenantConfig);
142       tenant.setTenant(tenantName);
143 
144       for (String tenantRoleName : tenant.getRoles()) {
145         List<Tenant> tenants = tenantsByRole.get(tenantRoleName);
146         if (tenants == null) {
147           tenants = new ArrayList<>();
148           tenantsByRole.put(tenantRoleName, tenants);
149         }
150         tenants.add(tenant);
151       }
152     }
153     map.put(TENANTS_BY_ROLE, tenantsByRole);
154 
155     return map;
156   }
157 
158   /**
159    * Build context variables specific for a node and role/variant.
160    * @param node Node
161    * @param nodeRole Node role
162    * @return Context variables map
163    */
164   public static Map<String, Object> buildCurrentContextVariables(Node node, NodeRole nodeRole) {
165     Map<String, Object> map = new HashMap<>();
166     map.put(ROLE, nodeRole.getRole());
167 
168     List<String> variants = nodeRole.getAggregatedVariants();
169     if (variants.size() == 1) {
170       map.put(ROLE_VARIANT, variants.get(0));
171     }
172     else {
173       map.put(ROLE_VARIANT, null);
174     }
175     map.put(ROLE_VARIANTS, variants);
176 
177     map.put(NODE, node.getNode());
178     return map;
179   }
180 
181   /**
182    * Removes all context variables.
183    * @param config Configuration
184    * @return Configuration
185    */
186   public static Map<String, Object> removeContextVariables(Map<String, Object> config) {
187     Map<String, Object> map = new HashMap<>(config);
188     map.remove(VERSION);
189     map.remove(ENVIRONMENT);
190     map.remove(NODES);
191     map.remove(NODES_BY_ROLE);
192     map.remove(NODES_BY_ROLE_VARIANT);
193     map.remove(TENANTS);
194     map.remove(TENANTS_BY_ROLE);
195     map.remove(ROLE);
196     map.remove(ROLE_VARIANT);
197     map.remove(NODE);
198     map.remove(TENANT);
199     map.remove(TENANT_ROLES);
200     return map;
201   }
202 
203   /**
204    * Get map with all context variables set to empty values.
205    * @return Map with context variables
206    */
207   public static Map<String, Object> getEmptyContextVariables() {
208     return EMPTY_CONTEXT_VARIABLES;
209   }
