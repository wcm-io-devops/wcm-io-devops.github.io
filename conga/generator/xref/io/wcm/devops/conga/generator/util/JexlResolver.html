20  package io.wcm.devops.conga.generator.util;
21  
22  import static io.wcm.devops.conga.generator.util.VariableStringResolver.EXPRESSION_POS_DOLLAR_SIGN;
23  import static io.wcm.devops.conga.generator.util.VariableStringResolver.EXPRESSION_POS_EXPRESSION;
24  import static io.wcm.devops.conga.generator.util.VariableStringResolver.MULTI_EXPRESSION_PATTERN;
25  import static io.wcm.devops.conga.generator.util.VariableStringResolver.VARIABLE_PATTERN;
26  
27  import java.util.List;
28  import java.util.Map;
29  import java.util.regex.Matcher;
30  
31  import org.apache.commons.jexl3.JexlBuilder;
32  import org.apache.commons.jexl3.JexlContext;
33  import org.apache.commons.jexl3.JexlEngine;
34  import org.apache.commons.jexl3.JexlException;
35  import org.apache.commons.jexl3.JexlExpression;
36  import org.apache.commons.jexl3.MapContext;
37  import org.apache.commons.lang3.StringUtils;
38  
39  import com.google.common.collect.ImmutableMap;
40  
41  import io.wcm.devops.conga.generator.GeneratorException;
42  import io.wcm.devops.conga.model.util.MapMerger;
43  import io.wcm.devops.conga.model.util.MapSplitter;
44  import io.wcm.devops.conga.model.util.MapSplitter.SplitResult;
45  
46  final class JexlResolver {
47  
48    private static final int CACHE_SIZE = 1024;
49  
50    private final JexlEngine jexl;
51    private final VariableMapResolver variableMapResolver;
52  
53    private static final Map<String, Object> JEXL_FUNCTION_NS = ImmutableMap.of(
54        "stringUtils", StringUtils.class);
55  
56    JexlResolver(VariableMapResolver variableMapResolver) {
57      this.jexl = new JexlBuilder()
58          .cache(CACHE_SIZE)
59          .namespaces(JEXL_FUNCTION_NS)
60          .create();
61      this.variableMapResolver = variableMapResolver;
62    }
63  
64    public Object resolve(String expressionString, Map<String, Object> variables) {
65      Map<String, Object> resolvedVariables = resolveMapWithoutCycles(variables);
66      try {
67        JexlExpression expression = jexl.createExpression(expressionString);
68        JexlContext context = new MapContext(resolvedVariables);
69        return expression.evaluate(context);
70      }
71      catch (JexlException ex) {
72        throw new GeneratorException("Unable to parse expression: ${" + expressionString + "}: " + ex.getMessage(), ex);
73      }
74    }
75  
76    private Map<String, Object> resolveMapWithoutCycles(Map<String, Object> variables) {
77  
78      // split out variables with values containing expressions that are not simple variable references
79      SplitResult splitResult = MapSplitter.splitMap(variables, entry -> {
80        if (entry.getValue() instanceof List) {
81          for (Object item : (List)entry.getValue()) {
82            if (item instanceof String) {
83              if (hasJexlExpresssions((String)item)) {
84                return false;
85              }
86            }
87          }
88        }
89        else if (entry.getValue() instanceof String) {
90          return !hasJexlExpresssions((String)entry.getValue());
91        }
92        return true;
93      });
94  
95      // resolve simple variable references inside map
96      Map<String, Object> resolvedVariables = variableMapResolver.resolve(splitResult.getMatching());
97  
98      // add back variables with complex expressions without resolving them
99      resolvedVariables = MapMerger.merge(resolvedVariables, splitResult.getUnmatching());
100 
101     return resolvedVariables;
102   }
103 
104   private boolean hasJexlExpresssions(String expressionString) {
105     Matcher matcher = MULTI_EXPRESSION_PATTERN.matcher(expressionString);
106     while (matcher.find()) {
107 
108       boolean escapedVariable = StringUtils.equals(matcher.group(EXPRESSION_POS_DOLLAR_SIGN), "\\$");
109       String expression = matcher.group(EXPRESSION_POS_EXPRESSION);
110 
111       // keep escaped variables intact
112       if (!escapedVariable) {
113         Matcher variableMatcher = VARIABLE_PATTERN.matcher(expression);
114         // if expression found that matches not the variable pattern this string contains Jexl expressions
115         if (!variableMatcher.matches()) {
116           return true;
117         }
118       }
119 
120     }
121     return false;
122   }
123 
124 }
