20  package io.wcm.devops.conga.tooling.maven.plugin.validation;
21  
22  import java.nio.charset.StandardCharsets;
23  
24  import org.apache.commons.lang3.StringUtils;
25  import org.apache.commons.lang3.Strings;
26  import org.apache.maven.plugin.MojoFailureException;
27  
28  import com.github.jknack.handlebars.Handlebars;
29  
30  import io.wcm.devops.conga.generator.handlebars.HandlebarsManager;
31  import io.wcm.devops.conga.generator.plugins.handlebars.escaping.NoneEscapingStrategy;
32  import io.wcm.devops.conga.resource.Resource;
33  import io.wcm.devops.conga.resource.ResourceCollection;
34  import io.wcm.devops.conga.tooling.maven.plugin.util.PathUtil;
35  
36  /**
37   * Validates Handlebars templates by compiling it.
38   */
39  public final class TemplateValidator implements DefinitionValidator<Void> {
40  
41    private static final String FILE_EXTENSION = "hbs";
42  
43    private final ResourceCollection templateDir;
44    private final HandlebarsManager handlebarsManager;
45  
46    /**
47     * Constructor.
48     * @param templateDir Template directory
49     * @param handlebarsManager Handlebars Manager
50     */
51    public TemplateValidator(ResourceCollection templateDir, HandlebarsManager handlebarsManager) {
52      this.templateDir = templateDir;
53      this.handlebarsManager = handlebarsManager;
54    }
55  
56    @Override
57    @SuppressWarnings({
58        "PMD.PreserveStackTrace",
59        "java:S1075" // uses / by intention
60    })
61    public Void validate(Resource resource, String pathForLog) throws MojoFailureException {
62      if (Strings.CI.equals(resource.getFileExtension(), FILE_EXTENSION)) {
63        String templatePath = StringUtils.substringAfter(PathUtil.unifySlashes(resource.getCanonicalPath()),
64            PathUtil.unifySlashes(templateDir.getCanonicalPath()) + "/");
65        Handlebars handlebars = handlebarsManager.get(NoneEscapingStrategy.NAME, StandardCharsets.UTF_8.name());
66        try {
67          handlebars.compile(templatePath);
68        }
69        /*CHECKSTYLE:OFF*/ catch (Exception ex) { /*CHECKSTYLE:ON*/
70          throw new MojoFailureException("Template " + pathForLog + " is invalid:\n" + ex.getMessage());
71        }
72      }
73      return null;
74    }
75  
76  }
