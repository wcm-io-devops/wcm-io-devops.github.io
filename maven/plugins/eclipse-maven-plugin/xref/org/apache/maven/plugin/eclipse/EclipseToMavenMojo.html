78  /**
79   * <p>Add Eclipse artifacts from an Eclipse installation or a local P2 repo to the local Maven repo.</p>
80   * 
81   * <p>This mojo automatically analyzes the local P2 repo, copies jars from the plugins directory to the local maven
82   * repo, and generates appropriate poms.</p>
83   * 
84   * <p>Typical usage for a local Eclipse installation:</p>
85   * <p><code>mvn eclipse:to-maven 
86   * -DdeployTo=maven.org::default::scpexe://repo1.maven.org/home/maven/repository-staging/to-ibiblio/eclipse-staging
87   * -DeclipseDir=/opt/eclipse</code></p>
88   * 
89   * <p>Or to download and convert a P2 repository:</p>
90   * <p><code>/opt/eclipse/eclipse -application org.eclipse.equinox.p2.artifact.repository.mirrorApplication
91   * -writeMode clean -verbose -raw -ignoreErrors -source http://download.eclipse.org/releases/oxygen
92   * -destination file:/tmp/p2-oxygen<br>
93   * mvn eclipse:to-maven -DbundleNameAsArtifactId=true -DgroupIdTokens=3 -DattachSourcePlugins
94   * -DresolveVersionRanges -DdeployTo=oxygen::default::file:/tmp/repo-2018-12 -DeclipseDir=/tmp/oxygen</code></p>
95   * 
96   * <p>Note: The size of such an update site (P2 repository) is 3 to 4 GB and the download can take several hours.</p>
97   * 
98   * @author Fabrizio Giustina
99   * @author <a href="mailto:carlos@apache.org">Carlos Sanchez</a>
100  * @version $Id$
101  */

223     /**
224      * @see org.apache.maven.plugin.Mojo#execute()
225      */
226     public void execute()
227         throws MojoExecutionException, MojoFailureException
228     {
229         if ( eclipseDir == null )
230         {
231             getLog().info( Messages.getString( "EclipseToMavenMojo.eclipseDirectoryPrompt" ) );
232 
233             String eclipseDirString;
234             try
235             {
236                 eclipseDirString = inputHandler.readLine();
237             }
238             catch ( IOException e )
239             {
240                 throw new MojoFailureException( 
241                                         Messages.getString( "EclipseToMavenMojo.errorreadingfromstandardinput" ) );
242             }
243             eclipseDir = new File( eclipseDirString );
244         }
245 
246         if ( !eclipseDir.isDirectory() )
247         {
248             throw new MojoFailureException( Messages.getString( "EclipseToMavenMojo.directoydoesnotexist",
249                                                                 eclipseDir.getAbsolutePath() ) );
250         }
251 
252         File pluginDir = new File( eclipseDir, "plugins" );
253 
254         if ( !pluginDir.isDirectory() )
255         {
256             throw new MojoFailureException( Messages.getString( "EclipseToMavenMojo.plugindirectorydoesnotexist",
257                                                                 pluginDir.getAbsolutePath() ) );
258         }
259 
260         File[] files = pluginDir.listFiles();
261 
262         ArtifactRepository remoteRepo = resolveRemoteRepo();
263 
264         if ( remoteRepo != null )
265         {
266             getLog().info( Messages.getString( "EclipseToMavenMojo.remoterepositorydeployto", deployTo ) );
267         }
268 
269         Map<String, EclipseOsgiPlugin> plugins = new HashMap<String, EclipseOsgiPlugin>();
270         Map<String, Model> models = new HashMap<String, Model>();
271 
272         getLog().info( Messages.getString( "EclipseToMavenMojo.searchingplugins", pluginDir.getAbsolutePath() ) );
273 
274         for ( File file : files )
275         {
276             getLog().debug( Messages.getString( "EclipseToMavenMojo.processingfile", file.getAbsolutePath() ) );
277 
278             processFile( file, plugins, models );
279         }
280 
281         getLog().info( Messages.getString( "EclipseToMavenMojo.pluginsfound", 
282             new Object[] { pluginDir.getAbsolutePath(), plugins.size() } ) );
283 
284         if ( attachSourcePlugins )
285         {
286             Set<String> sourceKeys = new HashSet<String>();
287             for ( String key : plugins.keySet() )
288             {
289                 if ( SOURCE_PATTERN.matcher( key ).matches() )
290                 {
291                     sourceKeys.add( key );
292                 }
293             }
294 
295             models.keySet().removeAll( sourceKeys );
296 
297             getLog().info( Messages.getString( "EclipseToMavenMojo.attachedsourceplugins", sourceKeys.size() ) );
298         }
299 
300         resolveVersions( models );
301 
302         getLog().info( Messages.getString( "EclipseToMavenMojo.deploymainartifacts", models.size() ) );
303 
304         int i = 1;
305         try
306         {
307             for ( String key : models.keySet() )
308             {
309                 getLog().debug( Messages.getString( "EclipseToMavenMojo.processingplugin", new Object[] { i++,
310                                    models.keySet().size() } ) );
311                 Model model = (Model) models.get( key );
312                 writeArtifact( model, plugins, remoteRepo );
313             }
314         }
315         finally
316         {
317             getLog().info( Messages.getString( "EclipseToMavenMojo.deployedmainartifacts", i ) );
318         }
319     }

321     protected void processFile( File file, Map<String, EclipseOsgiPlugin> plugins, Map<String, Model> models )
322         throws MojoExecutionException, MojoFailureException
323     {
324         EclipseOsgiPlugin plugin = getEclipsePlugin( file );
325 
326         if ( plugin == null )
327         {
328             getLog().warn( Messages.getString( "EclipseToMavenMojo.skippingfile", file.getAbsolutePath() ) );
329             return;
330         }
331 
332         Model model = createModel( plugin );
333 
334         if ( model == null )
335         {
336             return;
337         }
338 
339         processPlugin( plugin, model, plugins, models );
340     }

370     /**
371      * Resolve dependency versions in the models, that did not declare a version at all.
372      *
373      * @param models
374      * @throws MojoFailureException
375      */
376     protected void resolveVersions( Map<String, Model> models )
377         throws MojoFailureException
378     {
379         Map<String, SortedSet<ArtifactVersion>> allVersions = new HashMap<String, SortedSet<ArtifactVersion>>();
380         for ( Model model : models.values() )
381         {
382             String key = getModuleKey( model );
383             SortedSet<ArtifactVersion> versions = allVersions.get( key );
384             if ( versions == null )
385             {
386                 versions = new TreeSet<ArtifactVersion>( Collections.reverseOrder() );
387                 allVersions.put( key, versions );
388             }
389             versions.add( new DefaultArtifactVersion( model.getVersion() ) );
390         }
391 
392         for ( Model model : models.values() )
393         {
394             for ( Dependency dep : model.getDependencies() )
395             {
396                 try
397                 {
398                     String versionRange = dep.getVersion();
399                     if ( ANY_VERSION.equals( versionRange ) )
400                     {
401                         String key = getModuleKey( dep );
402                         Set<ArtifactVersion> versions = allVersions.get( key );
403                         if ( versions != null )
404                         {
405                             dep.setVersion( versions.iterator().next().toString() );
406                         }
407                     }
408                     else if ( "[(".indexOf( versionRange.charAt( 0 ) ) == 0 )
409                     {
410                         if ( resolveVersionRanges )
411                         {
412                             String key = getModuleKey( dep );
413                             Set<ArtifactVersion> versions = allVersions.get( key );
414                             if ( versions != null )
415                             {
416                                 selectVersion( dep, versionRange, versions );
417                             }
418                         }
419                     }
420                     else if ( resolveRecommendedVersions )
421                     {
422                         String key = getModuleKey( dep );
423                         Set<ArtifactVersion> versions = allVersions.get( key );
424                         ArtifactVersion version = new DefaultArtifactVersion( versionRange );
425                         if ( versions != null )
426                         {
427                             if ( !versions.contains( version ) )
428                             {
429                                 String lowerBound = String.format( "[%d.%d.%d,",
430                                     version.getMajorVersion(),
431                                     version.getMinorVersion(),
432                                     version.getIncrementalVersion() );
433                                 for ( int i = 3; i > 0; --i )
434                                 {
435                                     String upperBound = null;
436                                     switch ( i )
437                                     {
438                                         case 3:
439                                             upperBound = String.format( "%d.%d.%d)",
440                                                 version.getMajorVersion(),
441                                                 version.getMinorVersion(),
442                                                 version.getIncrementalVersion() + 1 );
443                                             break;
444 
445                                         case 2:
446                                             upperBound = String.format( "%d.%d)",
447                                                 version.getMajorVersion(),
448                                                 version.getMinorVersion() + 1 );
449                                             break;
450 
451                                         case 1:
452                                             upperBound = String.format( "%d)", version.getMajorVersion() + 1 );
453                                             break;
454 
455                                         default: break; // satisfy checkstyle
456                                     }
457 
458                                     if ( selectVersion( dep, lowerBound + upperBound, versions ) )
459                                     {
460                                         break;
461                                     }
462                                 }
463                             }
464                         }
465                     }
466                 }
467                 catch ( InvalidVersionSpecificationException e )
468                 {
469                     throw new MojoFailureException(
470                             Messages.getString( "EclipseToMavenMojo.invalidversionrange",
471                                 new Object[] { dep, getKey( model ) } ) );
472                 }
473             }
474         }
475     }
476 
477     private boolean selectVersion( Dependency dependency, String versionRange, Set<ArtifactVersion> versions )
478         throws InvalidVersionSpecificationException
479     {
480         VersionRange range = VersionRange.createFromVersionSpec( versionRange );
481         for ( ArtifactVersion version : versions )
482         {
483             if ( range.containsVersion( version ) )
484             {
485                 dependency.setVersion( version.toString() );
486                 return true;
487             }
488         }
489         return false;
490     }

492     /**
493      * Get a {@link EclipseOsgiPlugin} object from a plugin jar/dir found in the target dir.
494      *
495      * @param file plugin jar or dir
496      * @throws MojoExecutionException if anything bad happens while parsing files
497      */
498     private EclipseOsgiPlugin getEclipsePlugin( File file )
499         throws MojoExecutionException
500     {
501         if ( file.isDirectory() )
502         {
503             return new ExplodedPlugin( file );
504         }
505         else if ( file.getName().endsWith( ".jar" ) ) 
506         {
507             try
508             {
509                 return new PackagedPlugin( file );
510             }
511             catch ( IOException e )
512             {
513                 throw new MojoExecutionException( Messages.getString( "EclipseToMavenMojo.unabletoaccessjar",
514                                                                       file.getAbsolutePath() ), e );
515             }
516         }
517 
518         return null;
519     }

521     /**
522      * Create the {@link Model} from a plugin manifest
523      *
524      * @param plugin Eclipse plugin jar or dir
525      * @throws MojoExecutionException if anything bad happens while parsing files
526      */
527     private Model createModel( EclipseOsgiPlugin plugin )
528         throws MojoExecutionException
529     {
530 
531         String name, bundleName, version, groupId, artifactId, requireBundle;
532 
533         try
534         {
535             if ( !plugin.hasManifest() )
536             {
537                 getLog().warn( Messages.getString( "EclipseToMavenMojo.plugindoesnothavemanifest", plugin ) ); 
538                 return null;
539             }
540 
541             Analyzer analyzer = new Analyzer();
542 
543             Map bundleSymbolicNameHeader =
544                 analyzer.parseHeader( plugin.getManifestAttribute( Analyzer.BUNDLE_SYMBOLICNAME ) );
545             bundleName = (String) bundleSymbolicNameHeader.keySet().iterator().next();
546             version = plugin.getManifestAttribute( Analyzer.BUNDLE_VERSION );
547 
548             if ( bundleName == null || version == null )
549             {
550                 getLog().error( Messages.getString( "EclipseToMavenMojo.unabletoreadbundlefrommanifest" ) ); 
551                 return null;
552             }
553 
554             version = osgiVersionToMavenVersion( version );
555 
556             name = plugin.getManifestAttribute( Analyzer.BUNDLE_NAME );
557 
558             requireBundle = plugin.getManifestAttribute( Analyzer.REQUIRE_BUNDLE );
559 
560         }
561         catch ( IOException e )
562         {
563             throw new MojoExecutionException( Messages.getString( "EclipseToMavenMojo.errorprocessingplugin", plugin ),
564                                               e ); //$NON-NLS-1$
565         }
566 
567         Dependency[] deps = parseDependencies( requireBundle );
568 
569         groupId = createGroupId( bundleName );
570         artifactId = createArtifactId( bundleName );
571 
572         Model model = new Model();
573         model.setModelVersion( "4.0.0" ); 
574         model.setGroupId( groupId );
575         model.setArtifactId( artifactId );
576         model.setName( name );
577         model.setVersion( version );
578 
579         model.setProperties( plugin.getPomProperties() );
580 
581         if ( groupId.startsWith( "org.eclipse" ) ) 
582         {
583             // why do we need a parent?
584 
585             // Parent parent = new Parent();
586             // parent.setGroupId( "org.eclipse" );
587             // parent.setArtifactId( "eclipse" );
588             // parent.setVersion( "1" );
589             // model.setParent( parent );
590 
591             // infer license for know projects, everything at eclipse is licensed under EPL
592             // maybe too simplistic, but better than nothing
593             License license = new License();
594             license.setName( "Eclipse Public License - v 1.0" ); 
595             license.setUrl( "http://www.eclipse.org/org/documents/epl-v10.html" ); 
596             model.addLicense( license );
597         }
598 
599         if ( deps.length > 0 )
600         {
601             for ( Dependency dep : deps )
602             {
603                 model.getDependencies().add( dep );
604             }
605 
606         }
607 
608         return model;
609     }

611     /**
612      * Writes the artifact to the repo
613      *
614      * @param model
615      * @param remoteRepo remote repository (if set)
616      * @throws MojoExecutionException
617      */
618     private void writeArtifact( Model model, Map<String, EclipseOsgiPlugin> plugins, ArtifactRepository remoteRepo )
619         throws MojoExecutionException
620     {
621         Writer fw = null;
622         ArtifactMetadata metadata;
623         File pomFile = null;
624         Artifact pomArtifact =
625             artifactFactory.createArtifact( model.getGroupId(), model.getArtifactId(), model.getVersion(), null, 
626                                             "pom" );
627         Artifact artifact =
628             artifactFactory.createArtifact( model.getGroupId(), model.getArtifactId(), model.getVersion(), null,
629                                             Constants.PROJECT_PACKAGING_JAR );
630         Artifact sourcesArtifact = !attachSourcePlugins
631             ? null
632             :  artifactFactory.createArtifactWithClassifier( model.getGroupId(), model.getArtifactId(),
633                 model.getVersion(), Constants.PROJECT_PACKAGING_JAR, "sources" );
634         try
635         {
636             pomFile = File.createTempFile( "pom-", ".xml" ); 
637 
638             fw = WriterFactory.newWriter( pomFile, "UTF-8" );
639             model.setModelEncoding( "UTF-8" ); // to be removed when encoding is detected instead of forced to UTF-8 
640             pomFile.deleteOnExit();
641             new MavenXpp3Writer().write( fw, model );
642             metadata = new ProjectArtifactMetadata( pomArtifact, pomFile );
643             pomArtifact.addMetadata( metadata );
644         }
645         catch ( IOException e )
646         {
647             throw new MojoExecutionException( Messages.getString( "EclipseToMavenMojo.errorwritingtemporarypom",
648                                                                   e.getMessage() ), e );
649         }
650         finally
651         {
652             IOUtil.close( fw );
653         }
654 
655         EclipseOsgiPlugin plugin = plugins.get( getKey( model ) );
656         EclipseOsgiPlugin sourcePlugin = attachSourcePlugins ? plugins.get( getSourceKey( model ) ) : null;
657         File jarFile = null;
658         File jarFileSource = null;
659 
660         try
661         {
662             jarFile = plugin.getJarFile();
663             jarFileSource = sourcePlugin != null ? sourcePlugin.getJarFile() : null;
664 
665             if ( remoteRepo != null )
666             {
667                 deployer.deploy( pomFile, pomArtifact, remoteRepo, localRepository );
668                 deployer.deploy( jarFile, artifact, remoteRepo, localRepository );
669                 if ( sourcePlugin != null )
670                 {
671                     deployer.deploy( jarFileSource, sourcesArtifact, remoteRepo, localRepository );
672                 }
673             }
674             else
675             {
676                 installer.install( pomFile, pomArtifact, localRepository );
677                 installer.install( jarFile, artifact, localRepository );
678                 if ( sourcePlugin != null )
679                 {
680                     installer.install( jarFileSource, sourcesArtifact, localRepository );
681                 }
682             }
683         }
684         catch ( ArtifactDeploymentException e )
685         {
686             throw new MojoExecutionException( 
687                                   Messages.getString( "EclipseToMavenMojo.errordeployartifacttorepository" ), e );
688         }
689         catch ( ArtifactInstallationException e )
690         {
691             throw new MojoExecutionException( 
692                                   Messages.getString( "EclipseToMavenMojo.errorinstallartifacttorepository" ), e );
693         }
694         catch ( IOException e )
695         {
696             throw new MojoExecutionException(

698                                       jarFile != null ? sourcePlugin :  plugin ), e );
699         }
700         finally
701         {
702             pomFile.delete();

712     /**
713      * The 4th (build) token MUST be separated with "-" and not with "." in maven. A version with 4 dots is not parsed,
714      * and the whole string is considered a qualifier. See tests in DefaultArtifactVersion for reference.
715      *
716      * @param version initial version
717      * @param forcedQualifier build number
718      * @param stripQualifier always remove 4th token in version
719      * @return converted version
720      */
721     protected String osgiVersionToMavenVersion( String version, String forcedQualifier, boolean stripQualifier )
722     {
723         if ( stripQualifier && StringUtils.countMatches( version, "." ) > 2 )
724         {
725             version = StringUtils.substring( version, 0, version.lastIndexOf( '.' ) );
726         }
727         else if ( StringUtils.countMatches( version, "." ) > 2 )
728         {
729             int lastDot = version.lastIndexOf( '.' );
730             if ( StringUtils.isNotEmpty( forcedQualifier ) )
731             {
732                 version = StringUtils.substring( version, 0, lastDot ) + "-" + forcedQualifier;
733             }
734             else
735             {
736                 version = StringUtils.substring( version, 0, lastDot ) + "-" 
737                     + StringUtils.substring( version, lastDot + 1, version.length() );
738             }
739         }
740         return new DefaultArtifactVersion( version ).toString();
741     }

743     /**
744      * Resolves the deploy<code>deployTo</code> parameter to an <code>ArtifactRepository</code> instance (if set).
745      *
746      * @return ArtifactRepository instance of null if <code>deployTo</code> is not set.
747      * @throws MojoFailureException
748      * @throws MojoExecutionException
749      */
750     private ArtifactRepository resolveRemoteRepo()
751         throws MojoFailureException, MojoExecutionException
752     {
753         if ( deployTo != null )
754         {
755             Matcher matcher = DEPLOYTO_PATTERN.matcher( deployTo );
756 
757             if ( !matcher.matches() )
758             {
759                 throw new MojoFailureException( deployTo,
760                                             Messages.getString( "EclipseToMavenMojo.invalidsyntaxforrepository" ),
761                                             Messages.getString( "EclipseToMavenMojo.invalidremoterepositorysyntax" ) );
762             }
763             else
764             {
765                 String id = matcher.group( 1 ).trim();
766                 String layout = matcher.group( 2 ).trim();
767                 String url = matcher.group( 3 ).trim();
768 
769                 ArtifactRepositoryLayout repoLayout;
770                 try
771                 {
772                     repoLayout = (ArtifactRepositoryLayout) container.lookup( ArtifactRepositoryLayout.ROLE, layout );
773                 }
774                 catch ( ComponentLookupException e )
775                 {
776                     throw new MojoExecutionException(
777                                               Messages.getString( "EclipseToMavenMojo.cannotfindrepositorylayout",
778                                                                           layout ), e ); 
779                 }
780 
781                 return artifactRepositoryFactory.createDeploymentArtifactRepository( id, url, repoLayout, true );
782             }
783         }
784         return null;
785     }

796     /**
797      * Get the group id as the tokens until last dot e.g. <code>org.eclipse.jdt</code> -> <code>org.eclipse</code>
798      *
799      * @param bundleName bundle name
800      * @return group id
801      */
802     protected String createGroupId( String bundleName )
803     {
804         int i = 0;
805         int t = groupIdTokens;
806         if ( t < 0 )
807         {
808             i = bundleName.lastIndexOf( '.' ); //$NON-NLS-1$
809         }
810         else if ( t > 0 )
811         {
812             do
813             {
814                 i = bundleName.indexOf( '.', i + 1 ); //$NON-NLS-1$
815             } while ( --t > 0 && i > 0 );
816         }
817         if ( i > 0 )
818         {
819             return bundleName.substring( 0, i );
820         }
821         else
822         {
823             return bundleName;
824         }
825     }

827     /**
828      * Get the artifact id as the tokens after last dot e.g. <code>org.eclipse.jdt</code> -> <code>jdt</code>
829      *
830      * @param bundleName bundle name
831      * @return artifact id
832      */
833     protected String createArtifactId( String bundleName )
834     {
835         if ( !useBundleNameAsArtifactId )
836         {
837             String groupId = createGroupId( bundleName );
838             if ( bundleName.startsWith( groupId + '.' ) )  //$NON-NLS-1$
839             {
840                 return bundleName.substring( groupId.length() + 1 );
841             }
842         }
843         return bundleName;
844     }

846     /**
847      * Parses the "Require-Bundle" and convert it to a list of dependencies.
848      *
849      * @param requireBundle "Require-Bundle" entry
850      * @return an array of <code>Dependency</code>
851      */
852     protected Dependency[] parseDependencies( String requireBundle )
853     {
854         if ( requireBundle == null )
855         {
856             return new Dependency[0];
857         }
858 
859         List<Dependency> dependencies = new ArrayList<Dependency>();
860 
861         Analyzer analyzer = new Analyzer();
862 
863         @SuppressWarnings( "unchecked" )
864         Map<String, Map<String, String>> requireBundleHeader = analyzer.parseHeader( requireBundle );
865 
866         // now iterates on bundles and extract dependencies
867         for ( Map.Entry<String, Map<String, String>> entry : requireBundleHeader.entrySet() )
868         {
869             String bundleName = entry.getKey();
870             Map<String, String> attributes = entry.getValue();
871 
872             String version = attributes.get( Analyzer.BUNDLE_VERSION.toLowerCase() );
873             boolean optional = "optional".equals( attributes.get( "resolution:" ) );
874 
875             if ( version == null )
876             {
877                 getLog().debug( Messages.getString( "EclipseToMavenMojo.missingversionforbundle", bundleName ) );
878                 version = ANY_VERSION;
879             }
880 
881             version = fixBuildNumberSeparator( version );
882 
883             Dependency dep = new Dependency();
884             dep.setGroupId( createGroupId( bundleName ) );
885             dep.setArtifactId( createArtifactId( bundleName ) );
886             dep.setVersion( version );
887             dep.setOptional( optional );
888 
889             dependencies.add( dep );
890 
891         }
892 
893         return dependencies.toArray( new Dependency[dependencies.size()] );
894     }

896     /**
897      * Fix the separator for the 4th token in a versions. In maven this must be "-", in OSGI it's "."
898      *
899      * @param versionRange input range
900      * @return modified version range
901      */
902     protected String fixBuildNumberSeparator( String versionRange )
903     {
904         // should not be called with a null versionRange, but a check doesn't hurt...
905         if ( versionRange == null )
906         {
907             return null;
908         }
909 
910         StringBuffer newVersionRange = new StringBuffer();
911 
912         Matcher matcher = VERSION_PATTERN.matcher( versionRange );
913 
914         while ( matcher.find() )
915         {
916             String group = matcher.group();
917 
918             if ( StringUtils.countMatches( group, "." ) > 2 ) //$NON-NLS-1$
919             {
920                 // build number found, fix it
921                 int lastDot = group.lastIndexOf( '.' ); //$NON-NLS-1$
922                 group = StringUtils.substring( group, 0, lastDot ) + "-" //$NON-NLS-1$
923                     + StringUtils.substring( group, lastDot + 1, group.length() );
924             }
925             matcher.appendReplacement( newVersionRange, group );
926         }
927 
928         matcher.appendTail( newVersionRange );
929 
930         return newVersionRange.toString();
931     }
